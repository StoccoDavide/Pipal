238 bool optimize(Vector<Real> const & x_guess, Vector<Real> & x_sol)
239 {
240 #define CMD "Pipal::Solver::optimize(...): "
241
242 // Create alias for easier access
243 Parameter<Real> & p{this->m_parameter};
244 Counter & c{this->m_counter};
245 Input<Real> & i{this->m_input};
246 Iterate<Real> & z{this->m_iterate};
247 Direction<Real> & d{this->m_direction};
248 Acceptance<Real> & a{this->m_acceptance};
249
250 // Check that the problem is set
251 PIPAL_ASSERT(this->m_problem.get() != nullptr,
252 CMD "problem not set, use 'problem(...)' method to set it");
253
254 // Get variable bounds
255 Vector<Real> bl, bu;
256 PIPAL_ASSERT(this->m_problem->primal_lower_bounds(bl),
257 CMD "error in evaluating lower bounds on primal variables");
258 PIPAL_ASSERT(this->m_problem->primal_upper_bounds(bu),
259 CMD "error in evaluating upper bounds on primal variables");
260
261 // Get constraint bounds
262 Vector<Real> cl, cu;
263 PIPAL_ASSERT(this->m_problem->constraints_lower_bounds(cl),
264 CMD "error in evaluating lower bounds on constraints");
265 PIPAL_ASSERT(this->m_problem->constraints_upper_bounds(cu),
266 CMD "error in evaluating upper bounds on constraints");
267
268 // Reset counters
269 resetCounter(c);
270 resetDirection<Real>(d, i);
271
272 // Fill input structure
273 buildInput<Real>(i, p, this->m_problem->name(), x_guess, bl, bu, cl, cu);
274 buildIterate<Real>(z, p, i, c, *this->m_problem);
275
276 // Print header and break line
277 if (this->m_verbose) {this->m_output.printHeader(i, z); this->m_output.printBreak(c);}
278
279 // Iterations loop
280 while (!checkTermination<Real>(z, p, i, c))
281 {
282 // Print iterate
283 if (this->m_verbose) {this->m_output.printIterate(c, z);}
284
285 // Evaluate the step
286 evalStep<Real>(d, p, i, c, z, a, *this->m_problem);
287
288 // Print direction
289 if (this->m_verbose) {this->m_output.printDirection(z, d);}
290
291 lineSearch<Real>(a, p, i, c, z, d, *this->m_problem);
292
293 // Print accepted
294 if (this->m_verbose) {this->m_output.printAcceptance(a);}
295
296 updateIterate<Real>(z, p, i, c, d, a, *this->m_problem);
297
298 // Increment iteration counter
299 incrementIterationCount(c);
300
301 // Print break
302 if (this->m_verbose) {this->m_output.printBreak(c);}
303 }
304 // Print footer and terminate
305 if (this->m_verbose) {this->m_output.printFooter(p, i, c, z);}
306 x_sol = x_guess;
307 return true;
308
309 #undef CMD
310 }
