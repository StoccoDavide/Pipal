33
34 // Set number of original formulation variables
35 i.n0 = static_cast<Integer>(x0.size());
36
37 // Find indices sets
38 static constexpr Real EPSILON{std::numeric_limits<Real>::epsilon()};
39 Mask const cond_bl(bl.array() <= -p.rhs_bnd);
40 Mask const cond_bu(bu.array() >= p.rhs_bnd);
41 Mask const cond_cl(cl.array() <= -p.rhs_bnd);
42 Mask const cond_cu(cu.array() >= p.rhs_bnd);
43 Mask const cond_beq((bl.array() - bu.array()).abs() <= EPSILON);
44 Mask const cond_ceq((cl.array() - cu.array()).abs() <= EPSILON);
45
46 // Now the index sets
47 i.I1 = find(cond_bl && cond_bu);
48 i.I2 = find(cond_beq);
49 i.I3 = find(!cond_bl && cond_bu);
50 i.I4 = find(cond_bl && !cond_bu);
51 i.I5 = find(!cond_bl && !cond_bu && !cond_beq);
52 i.I6 = find(cond_ceq);
53 i.I7 = find(!cond_cl && cond_cu);
54 i.I8 = find(cond_cl && !cond_cu);
55 i.I9 = find(!cond_cl && !cond_cu && !cond_ceq);
56
57 // Set right-hand side values
58 i.b2 = bl(i.I2);
59 i.l3 = bl(i.I3);
60 i.u4 = bu(i.I4);
61 i.l5 = bl(i.I5);
62 i.u5 = bu(i.I5);
63 i.b6 = cl(i.I6);
64 i.l7 = cl(i.I7);
65 i.u8 = cu(i.I8);
66 i.l9 = cl(i.I9);
67 i.u9 = cu(i.I9);
68
69 // Set sizes of indices sets
70 i.n1 = static_cast<Integer>(i.I1.size());
71 i.n2 = static_cast<Integer>(i.I2.size());
72 i.n3 = static_cast<Integer>(i.I3.size());
73 i.n4 = static_cast<Integer>(i.I4.size());

75 i.n6 = static_cast<Integer>(i.I6.size());
76 i.n7 = static_cast<Integer>(i.I7.size());

79

81 i.vi = 0;
82
83 // Count invalid bounds
84 if (i.n2 > 0) {
85 i.vi += (i.b2.array() <= -p.rhs_bnd).count();
86 i.vi += (i.b2.array() >= p.rhs_bnd).count();
87 }
88 if (i.n3 > 0) {
89 i.vi += (i.l3.array() >= p.rhs_bnd).count();
90 }
91 if (i.n4 > 0) {
92 i.vi += (i.u4.array() <= -p.rhs_bnd).count();
93 }
94 if (i.n5 > 0) {
95 i.vi += (i.l5.array() >= p.rhs_bnd).count();
96 i.vi += (i.u5.array() <= -p.rhs_bnd).count();
97 i.vi += (i.l5.array() > i.u5.array()).count();
98 }
99 if (i.n6 > 0) {
100 i.vi += (i.b6.array() <= -p.rhs_bnd).count();
101 i.vi += (i.b6.array() >= p.rhs_bnd).count();
102 }
103 if (i.n7 > 0) {
104 i.vi += (i.l7.array() >= p.rhs_bnd).count();
105 }
106 if (i.n8 > 0) {
107 i.vi += (i.u8.array() <= -p.rhs_bnd).count();
108 }
109 if (i.n9 > 0) {
110 i.vi += (i.l9.array() >= p.rhs_bnd).count();
111 i.vi += (i.u9.array() <= -p.rhs_bnd).count();
112 i.vi += (i.l9.array() > i.u9.array()).count();
113 }
114
115 // Set number of variables and constraints
116 i.nV = i.n1 + i.n3 + i.n4 + i.n5;
117 i.nI = i.n3 + i.n4 + 2*i.n5 + i.n7 + i.n8 + 2*i.n9;
118 i.nE = i.n6;
119
120 // Set size of primal-dual matrix
121 i.nA = i.nV + 3*i.nE + 3*i.nI;
122
123 // Set initial point
124 i.x0.resize(i.nV);
125 i.x0 << x0(i.I1), x0(i.I3), x0(i.I4), x0(i.I5);

128 }
