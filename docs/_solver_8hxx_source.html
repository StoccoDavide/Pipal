383 bool optimize(Vector<Real> const & x_guess, Vector<Real> & x_sol)
384 {
385
386 #define CMD "Pipal::Solver::optimize(...): "
387
388 // Create alias for easier access
389 Counter & c{this->m_counter};
390 Input<Real> & i{this->m_input};
391 Iterate<Real> & z{this->m_iterate};
392 Direction<Real> & d{this->m_direction};
393 Acceptance<Real> & a{this->m_acceptance};
394
395 // Check that the problem is set
396 PIPAL_ASSERT(this->m_problem.get() != nullptr,
397 CMD "problem not set, use 'problem(...)' method to set it");
398
399 // Get variable bounds
400 Vector<Real> bl, bu;
401 PIPAL_ASSERT(this->m_problem->primal_lower_bounds(bl),
402 CMD "error in evaluating lower bounds on primal variables");
403 PIPAL_ASSERT(this->m_problem->primal_upper_bounds(bu),
404 CMD "error in evaluating upper bounds on primal variables");
405
406 // Get constraint bounds
407 Vector<Real> cl, cu;
408 PIPAL_ASSERT(this->m_problem->constraints_lower_bounds(cl),
409 CMD "error in evaluating lower bounds on constraints");
410 PIPAL_ASSERT(this->m_problem->constraints_upper_bounds(cu),
411 CMD "error in evaluating upper bounds on constraints");
412
413 // Reset counters
414 resetCounter();
415
416 // Fill input structure
417 buildInput(this->m_problem->name(), x_guess, bl, bu, cl, cu);
418 buildIterate();
419 resetDirection(d);
420
421 // Print header and break line
422 if (this->m_verbose) {this->m_output.printHeader(i, z); this->m_output.printBreak(c);}
423
424 // Iterations loop
425 while (!this->checkTermination()) {
426
427 // Print iterate
428 if (this->m_verbose) {this->m_output.printIterate(c, z);}
429
430 // Evaluate the step
431 this->evalStep();
432
433 // Print direction
434 if (this->m_verbose) {this->m_output.printDirection(z, d);}
435
436 this->lineSearch();
437
438 // Print accepted
439 if (this->m_verbose) {this->m_output.printAcceptance(a);}
440
441 this->updateIterate();
442
443 // Increment iteration counter
444 this->incrementIterationCount();
445
446 // Print break
447 if (this->m_verbose) {this->m_output.printBreak(c);}
448 }
449 // Print footer and terminate
450 if (this->m_verbose) {this->m_output.printFooter(c, z, this->checkTermination());}
451
452 // Get solution in original variables
453 this->evalXOriginal(x_sol);
454
455 // Return success if is finite
456 return x_sol.allFinite();
457
458 #undef CMD
459 }
