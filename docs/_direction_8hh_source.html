31 {
32 d.x.setZero(i.nV);
33 d.r1.setZero(i.nE);
34 d.r2.setZero(i.nE);

38 d.lI.setZero(i.nI);

57 template<typename Real>
58 inline void evalLinearCombination(Direction<Real> & d, Input<Real> const & i, Direction<Real> const & d1,
59 Direction<Real> const & d2, Direction<Real> const & d3, Real const a1, Real const a2, Real const a3)
60 {
61 // Evaluate linear combinations
62 d.x = a1*d1.x + a2*d2.x + a3*d3.x;
63 if (i.nE > 0) {
64 d.r1 = a1*d1.r1 + a2*d2.r1 + a3*d3.r1;
65 d.r2 = a1*d1.r2 + a2*d2.r2 + a3*d3.r2;
66 d.lE = a1*d1.lE + a2*d2.lE + a3*d3.lE;
67 }
68 if (i.nI > 0) {
69 d.s1 = a1*d1.s1 + a2*d2.s1 + a3*d3.s1;
70 d.s2 = a1*d1.s2 + a2*d2.s2 + a3*d3.s2;
71 d.lI = a1*d1.lI + a2*d2.lI + a3*d3.lI;
72 }
73
74 // Evaluate primal direction norm
75 d.x_norm = d.x.norm();
76
77 // Evaluate dual direction norm
78 d.l_norm = std::sqrt(d.lE.matrix().squaredNorm() + d.lI.matrix().squaredNorm());
79 }

88 template<typename Real>
89 inline void evalModels(Direction<Real> & d, Input<Real> const & i, Iterate<Real> & z)
90 {
91 // Evaluate reduction in linear model of penalty-interior-point objective for zero penalty parameter
92 d.lred0 = 0;
93 if (i.nE > 0) {
94 d.lred0 -= ((1.0 - z.mu / z.r1) * d.r1).sum();
95 d.lred0 -= ((1.0 - z.mu / z.r2) * d.r2).sum();
96 }
97 if (i.nI > 0) {
98 d.lred0 -= (((-z.mu) / z.s1) * d.s1).sum();
99 d.lred0 -= ((1.0 - z.mu / z.s2) * d.s2).sum();
100 }
101
102 // Evaluate remaining quantities only for nonzero penalty parameter
103 if (z.rho > 0)
104 {
105 // Evaluate reduction in linear model of merit function for zero penalty parameter
106 d.ltred0 = 0;
107 if (i.nE > 0) {
108 Array<Real> sqrt_term((z.cE.square() + z.mu*z.mu).sqrt());
109 d.ltred0 -= 0.5*(((1.0 - z.mu/z.r1) * (-1.0 + z.cE/sqrt_term) + (1.0 - z.mu/z.r2) *
110 (1.0 + z.cE/sqrt_term)) * (z.JE * d.x).array()).sum();
111 }
112 if (i.nI > 0) {
113 Array<Real> sqrt_term((z.cI.square() + 4.0*z.mu*z.mu).sqrt());
114 d.ltred0 -= 0.5*((((-z.mu)/z.s1) * (-1.0 + z.cI/sqrt_term) + (1.0 - z.mu/z.s2) *
115 (1.0 + z.cI/sqrt_term)) * (z.JI * d.x).array()).sum();
116 }
117
118 // Evaluate reduction in linear model of merit function
119 d.ltred = -z.rho*z.g.transpose()*d.x + d.ltred0;
120
121 // Evaluate reduction in quadratic model of merit function
122 d.qtred = d.ltred - 0.5*d.x.transpose()*z.H*d.x;
123 if (i.nE > 0) {
124 Array<Real> Jd(z.JE*d.x);
125 Array<Real> Dinv((z.r1/(1.0+z.lE) + z.r2/(1.0-z.lE)).matrix());
126 d.qtred -= 0.5*Jd.matrix().transpose() * ((Jd/Dinv).matrix());
127 }
128 if (i.nI > 0) {
129 Array<Real> Jd(z.JI*d.x);
130 Array<Real> Dinv((z.s1/(0.0+z.lI) + z.s2/(1.0-z.lI)).matrix());
131 d.qtred -= 0.5*Jd.matrix().transpose() * ((Jd/Dinv).matrix());
132 }
133
134 // Initialize quality function vector
135 Array<Real> vec(i.nV+2*i.nE+2*i.nI);
136 vec.setZero();
137
138 // Set gradient of objective
139 vec.head(i.nV) = z.rho*z.g;
140
141 // Set gradient of Lagrangian for constraints
142 if (i.nE > 0) {vec.head(i.nV) += ((z.lE+d.lE).matrix().transpose()*z.JE).transpose().array();}
143 if (i.nI > 0) {vec.head(i.nV) += ((z.lI+d.lI).matrix().transpose()*z.JI).transpose().array();}
144
145 // Set complementarity for constraint slacks
146 if (i.nE > 0) {
147 vec.segment(i.nV, 2*i.nE) <<
148 (z.r1+d.r1)*(1.0 + (z.lE+d.lE)), (z.r2+d.r2)*(1.0 - (z.lE+d.lE));
149 }
150 if (i.nI > 0) {
151 vec.segment(i.nV+2*i.nE, 2*i.nI) <<
152 (z.s1+d.s1)*(z.lI+d.lI), (z.s2+d.s2)*(1.0 - (z.lI+d.lI));
153 }
154
155 // Evaluate quality function
156 d.m = vec.matrix().template lpNorm<Eigen::Infinity>();
157 }
158 }

167 template<typename Real>
168 inline void evalNewtonStep(Direction<Real> & d, Input<Real> const & i, Iterate<Real> const & z)
169 {
170 // Evaluate direction
171 SparseVector<Real> dir(z.ldlt.solve(-z.b));
172
173 // Parse direction
174 d.x = dir.head(i.nV);
175 if (i.nE > 0) {
176 d.r1 = dir.segment(i.nV, i.nE);
177 d.r2 = dir.segment(i.nV+i.nE, i.nE);
178 d.lE = dir.segment(i.nV+i.nE+i.nE+i.nI+i.nI, i.nE);
179 }
180 if (i.nI > 0) {
181 d.s1 = dir.segment(i.nV+i.nE+i.nE, i.nI);
182 d.s2 = dir.segment(i.nV+i.nE+i.nE+i.nI, i.nI);
183 d.lI = dir.segment(i.nV+i.nE+i.nE+i.nI+i.nI+i.nE, i.nI);
184 }
185
186 // Evaluate primal direction norm
187 d.x_norm = d.x.norm();
188
189 // Evaluate dual direction norm
190 d.l_norm = std::sqrt(d.lE.matrix().squaredNorm() + d.lI.matrix().squaredNorm());
191 }

204 template<typename Real>
205 inline void evalStep(Direction<Real> & d, Parameter<Real> & p, Input<Real> & i, Counter & c,
206 Iterate<Real> & z, Acceptance<Real> & a, Problem<Real> const * problem)
207 {
208 // Reset maximum exponent for interior-point parameter increases
209 resetMuMaxExp(p);
210
211 // Update penalty-interior-point parameters based on KKT errors
212 updateParameters<Real>(z, p, i);
213
214 // Evaluate matrices
215 evalMatrices<Real>(z, p, i, c, problem);
216
217 // Set last penalty parameter
218 setRhoLast<Real>(z, z.rho);
219
220 // Check for aggressive algorithm
221 if (p.algorithm == Algorithm::ADAPTIVE)
222 {
223 // Check KKT memory for potential mu increase limit
224 if (z.kkt(1) > z.kkt_.maxCoeff()) {setMuMaxExpZero(p);}
225
226 // Store current penalty and interior-point parameters
227 Real rho_curr{z.rho}, mu_curr{z.mu};
228
229 // Evaluate trial steps
230 Direction<Real> d1, d2, d3;
231 resetDirection(d1, i);
232 resetDirection(d2, i);
233 resetDirection(d3, i);
234 evalTrialSteps<Real>(d, i, z, d1, d2, d3);
235
236 // Set trial interior-point parameter values
237 Array<Real> exponents(Array<Real>::LinSpaced(p.mu_trials, p.mu_trials - 1, 0) - p.mu_max_exp);
238 Array<Real> Mu((mu_curr * exponents.unaryExpr(
239 [&p] (Real e) {return std::pow(p.mu_factor, e);})
240 ).min(p.mu_max).max(p.mu_min));
241
242 // Initialize feasibility direction data
243 Vector<Real> lred0_0_mu(p.mu_trials);
244 lred0_0_mu.setZero();
245
246 // Loop through interior-point parameter values
247 for (Integer j{0}; j < p.mu_trials; ++j)
248 {
249 // Set penalty and interior-point parameters
250 setRho<Real>(z, 0.0);
251 setMu<Real>(z, Mu(j));
252
253 // Evaluate direction
254 evalLinearCombination<Real>(d, i, d1, d2, d3,
255 z.rho/rho_curr+z.mu/mu_curr-1.0,
256 1.0-z.mu/mu_curr,
257 1.0-z.rho/rho_curr);
258
259 // Cut length
260 d.x *= std::min(d.x_norm_/std::max(d.x_norm, 1.0), 1.0);
261
262 // Run fraction-to-boundary
263 fractionToBoundary<Real>(a, p, i, z, d);
264
265 // Cut length
266 evalTrialStepCut<Real>(d, i, a);
267
268 // Evaluate models
269 evalModels<Real>(d, i, z);
270
271 // Set feasibility direction data
272 lred0_0_mu(j) = d.lred0;
273 }
274
275 // Initialize updating data
276 Vector<Real> ltred0_rho_mu(p.mu_trials), qtred_rho_mu(p.mu_trials), m_rho_mu(p.mu_trials);
277 ltred0_rho_mu.setZero(); qtred_rho_mu.setZero(); m_rho_mu.setZero();
278
279 // Initialize check
280 bool check{false};
281
282 // Loop through penalty parameter values
283 for (Integer k{0}; k < p.rho_trials; ++k)
284 {
285 // Set penalty parameter
286 setRho<Real>(z, std::max(p.rho_min, std::pow(p.rho_factor, k)*rho_curr));
287
288 // Set last penalty parameter
289 if (rho_curr > z.kkt(0)*z.kkt(0)) {setRhoLast<Real>(z, z.rho);}
290
291 // Loop through interior-point parameter values
292 for (Integer j{0}; j < p.mu_trials; ++j)
293 {
294 // Set interior-point parameter
295 setMu<Real>(z, Mu(j));
296
297 // Evaluate direction
298 evalLinearCombination<Real>(d, i, d1, d2, d3,
299 z.rho/rho_curr+z.mu/mu_curr-1.0,
300 1.0-z.mu/mu_curr,
301 1.0-z.rho/rho_curr);
302
303 // Run fraction-to-boundary
304 fractionToBoundary<Real>(a, p, i, z, d);
305
306 // Cut steps
307 evalTrialStepCut<Real>(d, i, a);
308
309 // Evaluate models
310 evalModels<Real>(d, i, z);
311
312 // Set updating data
313 ltred0_rho_mu(j) = d.ltred0;
314 qtred_rho_mu(j) = d.qtred;
315 m_rho_mu(j) = d.m;
316
317 // Check updating conditions for infeasible points
318 if (z.v > p.opt_err_tol && (ltred0_rho_mu(j) < p.update_con_1*lred0_0_mu(j) ||
319 qtred_rho_mu(j) < p.update_con_2*lred0_0_mu(j) || z.rho > z.kkt(0)*z.kkt(0))) {
320 m_rho_mu(j) = std::numeric_limits<Real>::infinity();
321 }
322
323 // Check updating conditions for feasible points
324 if (z.v <= p.opt_err_tol && qtred_rho_mu(j) < 0.0) {
325 m_rho_mu(j) = std::numeric_limits<Real>::infinity();
326 }
327 }
328
329 // Find minimum m for current rho
330 Real m_min{m_rho_mu.minCoeff()};
331
332 // Check for finite minimum
333 if (m_min < std::numeric_limits<Real>::infinity())
334 {
335 // Loop through mu values
336 for (Integer j{0}; j < p.mu_trials; ++j)
337 {
338 // Check condition
339 if (m_rho_mu(j) <= p.update_con_3*m_min) {setMu<Real>(z, Mu(j));}
340 }
341
342 // Set condition check
343 check = true;
344
345 // Break loop
346 break;
347 }
348 }
349
350 // Check conditions
351 if (check == false) {setRho<Real>(z, rho_curr); setMu<Real>(z, mu_curr);}
352
353 // Evaluate merit
354 evalMerit<Real>(z, i);
355 }
356
357 // Evaluate primal-dual right-hand side vector
358 evalNewtonRhs<Real>(z, i);
359
360 // Evaluate search direction
361 evalNewtonStep<Real>(d, i, z);
362
363 // Evaluate models
364 evalModels<Real>(d, i, z);
365
366 // Store last direction norm
367 d.x_norm_ = d.x_norm;
368 }

412 template<typename Real>
413 inline void evalTrialSteps(Direction<Real> & d, Input<Real> const & i, Iterate<Real> & z,
414 Direction<Real> & d1, Direction<Real> & d2, Direction<Real> & d3)
415 {
416 // Store current penalty and interior-point parameters
417 Real rho_curr{z.rho}, mu_curr{z.mu};
418
419 // Evaluate direction for current penalty and interior-point parameters
420 setRho<Real>(z, rho_curr);
421 setMu<Real>(z, mu_curr);
422 evalNewtonRhs<Real>(z, i);
423 evalNewtonStep<Real>(d, i, z);
424 evalTrialStep<Real>(d, i, d1);
425
426 // Evaluate direction for zero interior-point parameter
427 setRho<Real>(z, rho_curr);
428 setMu<Real>(z, 0.0);
429 evalNewtonRhs<Real>(z, i);
430 evalNewtonStep<Real>(d, i, z);
431 evalTrialStep<Real>(d, i, d2);
432
433 // Evaluate direction for zero penalty parameter
434 setRho<Real>(z, 0.0);
435 setMu<Real>(z, mu_curr);
436 evalNewtonRhs<Real>(z, i);
437 evalNewtonStep<Real>(d, i, z);
438 evalTrialStep<Real>(d, i, d3);
439 }
