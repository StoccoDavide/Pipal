33 template <typename Real>
34 inline void buildIterate(Iterate<Real> & z, Parameter<Real> & p, Input<Real> & i, Counter & c,
35 Problem<Real> const * problem)
36 {
37 // Safe initialization of all members
38 z.f = 0;
39 z.fu = 0;
40 z.g.setZero(i.nV);
41 z.r1.setZero(i.nE);
42 z.r2.setZero(i.nE);
43 z.cE.setZero(i.nE);
44 z.JE.resize(i.nE, i.nV);
45 z.s1.setZero(i.nI);
46 z.s2.setZero(i.nI);
47 z.cI.setZero(i.nI);
48 z.JI.resize(i.nI, i.nV);
49 z.H.resize(i.nV, i.nV);
50 z.v = 0;
51 z.vu = 0;
52 z.phi = 0;
53 z.Annz = 0;
54 z.shift = 0;
55 z.b.resize(i.nA);
56 z.kkt.setZero(3);
57 z.kkt_.setConstant(p.opt_err_mem, std::numeric_limits<Real>::infinity());
58 z.fs = 1.0;
59 z.cEs.setOnes(i.nE);
60 z.cEu.setZero(i.nE);
61 z.cIs.setOnes(i.nI);
62 z.cIu.setZero(i.nI);
63 z.A.resize(i.nA, i.nA);
64 z.shift22 = 0;
65 z.cut_ = false;
66
67 // Initialize point
68 z.x = i.x0;
69 z.rho = p.rho_init;
70 z.mu = p.mu_init;
71 z.lE.setZero(i.nE);
72 z.lI.setConstant(i.nI, 0.5);
73 z.err = 0;
74 evalScalings<Real>(z, p, i, c, problem);
75 evalFunctions<Real>(z, i, c, problem);
76 evalGradients<Real>(z, i, c, problem);
77 evalDependent<Real>(z, p, i);
78 z.v0 = 1.0;
79 evalInfeasibility<Real>(z, i);
80 z.v0 = z.v;
81 evalInfeasibility<Real>(z, i);
82 z.v_ = z.v;
83 evalHessian<Real>(z, i, c, problem);
84 z.Hnnz = static_cast<Integer>(z.H.nonZeros());
85 z.JEnnz = static_cast<Integer>(z.JE.nonZeros());
86 z.JInnz = static_cast<Integer>(z.JI.nonZeros());
87 initNewtonMatrix<Real>(z, i);
88 evalNewtonMatrix<Real>(z, p, i, c);
89 }

99 template <typename Real>
100 inline Integer checkTermination(Iterate<Real> const & z, Parameter<Real> const & p, Input<Real> const & i,
101 Counter const & c)
102 {
103 // Update termination based on optimality error of nonlinear optimization problem
104 if (z.kkt(1) <= p.opt_err_tol && z.v <= p.opt_err_tol) {return 1;}
105
106 // Update termination based on optimality error of feasibility problem
107 if (z.kkt(0) <= p.opt_err_tol && z.v > p.opt_err_tol) {return 2;}
108
109 // Update termination based on iteration count
110 if (c.k >= p.iter_max) {return 3;}
111
112 // Update termination based on invalid bounds
113 if (i.vi > 0) {return 4;}
114
115 // Update termination based on function evaluation error
116 if (z.err > 0) {return 5;}
117
118 return 0;
119 }

147 template <typename Real>
148 inline void evalFunctions(Iterate<Real> & z, Input<Real> & i, Counter & c, Problem<Real> const * problem)
149 {
150 // Evaluate x in original space
151 Vector<Real> x_orig;
152 evalXOriginal<Real>(z, i, x_orig);
153
154 // Initialize/Reset evaluation flag
155 z.err = 0;
156
157 // Increment function evaluation counter
158 incrementFunctionCount(c);
159
160 // Try AMPL functions evaluation
161 Vector<Real> c_orig;
162 try
163 {
164 // Evaluate AMPL functions
165 problem->objective(x_orig, z.f);
166 problem->constraints(x_orig, c_orig);
167 }
168 catch (...)
169 {
170 // Set evaluation flag, default values, and return
171 z.err = 1;
172 z.f = std::numeric_limits<Real>::quiet_NaN();
173 z.cE.setConstant(i.nE, std::numeric_limits<Real>::quiet_NaN());
174 z.cI.setConstant(i.nI, std::numeric_limits<Real>::quiet_NaN());
175 z.fu = std::numeric_limits<Real>::quiet_NaN();
176 z.cEu.setConstant(i.nE, std::numeric_limits<Real>::quiet_NaN());
177 z.cIu.setConstant(i.nI, std::numeric_limits<Real>::quiet_NaN());
178 return;
179 }
180
181 // Set equality constraint values
182 if (i.nE > 0) {z.cE = c_orig(i.I6) - i.b6;}
183
184 // Initialize inequality constraint values
185 if (i.nI > 0) {z.cI.setZero(i.nI);}
186
187 // Set inequality constraint values
188 if (i.n3 > 0) {
189 z.cI.head(i.n3) = i.l3 - z.x.segment(i.n1, i.n3);
190 }
191 if (i.n4 > 0) {
192 z.cI.segment(i.n3, i.n4) = -i.u4 + z.x.segment(i.n1+i.n3, i.n4);
193 }
194 if (i.n5 > 0) {
195 z.cI.segment(i.n3+i.n4, i.n5) = i.l5 - z.x.segment(i.n1+i.n3+i.n4, i.n5);
196 z.cI.segment(i.n3+i.n4+i.n5, i.n5) = -i.u5 + z.x.segment(i.n1+i.n3+i.n4, i.n5);
197 }
198 if (i.n7 > 0) {
199 z.cI.segment(i.n3+i.n4+i.n5+i.n5, i.n7) = i.l7 - c_orig(i.I7);
200 }
201 if (i.n8 > 0) {
202 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7, i.n8) = -i.u8 + c_orig(i.I8);
203 }
204 if (i.n9 > 0) {
205 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8, i.n9) = i.l9 - c_orig(i.I9);
206 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8+i.n9, i.n9) = -i.u9 + c_orig(i.I9);
207 }
208
209 // Store unscaled quantities
210 z.fu = z.f;
211 if (i.nE > 0) {z.cEu = z.cE;}
212 if (i.nI > 0) {z.cIu = z.cI;}
213
214 // Scale quantities
215 z.f *= z.fs;
216 if (i.nE > 0) {z.cE = (z.cEs*z.cE).matrix();}
217 if (i.nI > 0) {z.cI = (z.cIs*z.cI).matrix();}
218
219 }

231 template <typename Real, bool CheckZero = true>
232 inline void insert_block(SparseMatrix<Real> & mat_sparse, Matrix<Real> const & mat_dense,
233 Integer const row_offset, Integer const col_offset)
234 {
235 const Integer cols{static_cast<Integer>(mat_dense.cols())}, rows{static_cast<Integer>(mat_dense.rows())};
236 for (Integer r{0}; r < rows; ++r) {
237 for (Integer c{0}; c < cols; ++c) {
238 if constexpr (CheckZero) {
239 if (mat_dense(r, c) != static_cast<Real>(0.0)) {
240 mat_sparse.coeffRef(r + row_offset, c + col_offset) = mat_dense(r, c);
241 }
242 } else {
243 mat_sparse.coeffRef(r + row_offset, c + col_offset) = mat_dense(r, c);
244 }
245 }
246 }

261 template <typename Real>
262 inline void evalGradients(Iterate<Real> & z, Input<Real> & i, Counter & c, Problem<Real> const * problem)
263 {
264 // Evaluate x in original space
265 Vector<Real> x_orig;
266 evalXOriginal<Real>(z, i, x_orig);
267
268 // Initialize/Reset evaluation flag
269 z.err = 0;
270
271 // Increment gradient evaluation counter
272 incrementGradientCount(c);
273
274 // Try AMPL gradients evaluation
275 Vector<Real> g_orig;
276 Matrix<Real> J_orig;
277 try
278 {
279 // Evaluate AMPL gradients
280 problem->objective_gradient(x_orig, g_orig);
281 problem->constraints_jacobian(x_orig, J_orig);
282 }
283 catch (...)
284 {
285 // Set evaluation flag, default values, and return
286 z.err = 1;
287 return;
288 }
289
290 // Set objective gradient
291 z.g << g_orig(i.I1), g_orig(i.I3), g_orig(i.I4), g_orig(i.I5);
292
293 // Set equality constraint Jacobian
294 if (i.nE > 0) {
295 Integer col_offset{0};
296 insert_block<Real>(z.JE, J_orig(i.I6, i.I1), 0, col_offset);
297 col_offset += i.I1.size();
298 insert_block<Real>(z.JE, J_orig(i.I6, i.I3), 0, col_offset);
299 col_offset += i.I3.size();
300 insert_block<Real>(z.JE, J_orig(i.I6, i.I4), 0, col_offset);
301 col_offset += i.I4.size();
302 insert_block<Real>(z.JE, J_orig(i.I6, i.I5), 0, col_offset);
303 }
304
305 // Initialize inequality constraint Jacobian
306 //if (i.nI > 0) {z.JI.resize(i.nI, i.nV);}
307
308 // Set inequality constraint Jacobian
309 if (i.n3 > 0) {
310 for (Integer k{0}; k < i.n3; ++k) {z.JI.coeffRef(k, i.n1+k) = -1.0;}
311 }
312 if (i.n4 > 0) {
313 Integer tmp{i.n1+i.n3};
314 for (Integer k{0}; k < i.n4; ++k) {z.JI.coeffRef(i.n3+k, tmp+k) = 1.0;}
315 }
316 if (i.n5 > 0) {
317 Integer tmp_row{i.n3+i.n4}, tmp_col{i.n1+i.n3+i.n4};
318 for (Integer k{0}; k < i.n5; ++k) {
319 z.JI.coeffRef(tmp_row+k, tmp_col+k) = -1.0;
320 z.JI.coeffRef(tmp_row+i.n5+k, tmp_col+k) = 1.0;
321 }
322 }
323 if (i.n7 > 0) {
324 Integer row_offset{i.n3+i.n4+i.n5+i.n5}, col_offset{0};
325 insert_block<Real>(z.JI, -J_orig(i.I7, i.I1), row_offset, col_offset);
326 col_offset += i.I1.size();
327 insert_block<Real>(z.JI, J_orig(i.I7, i.I3), row_offset, col_offset);
328 col_offset += i.I3.size();
329 insert_block<Real>(z.JI, J_orig(i.I7, i.I4), row_offset, col_offset);
330 col_offset += i.I4.size();
331 insert_block<Real>(z.JI, J_orig(i.I7, i.I5), row_offset, col_offset);
332 }
333 if (i.n8 > 0) {
334 Integer row_offset{i.n3+i.n4+i.n5+i.n5+i.n7}, col_offset{0};
335 insert_block<Real>(z.JI, J_orig(i.I8, i.I1), row_offset, col_offset);
336 col_offset += i.I1.size();
337 insert_block<Real>(z.JI, J_orig(i.I8, i.I3), row_offset, col_offset);
338 col_offset += i.I3.size();
339 insert_block<Real>(z.JI, J_orig(i.I8, i.I4), row_offset, col_offset);
340 col_offset += i.I4.size();
341 insert_block<Real>(z.JI, J_orig(i.I8, i.I5), row_offset, col_offset);
342 }
343 if (i.n9 > 0) {
344 Integer row_offset{i.n3+i.n4+i.n5+i.n5+i.n7+i.n8}, col_offset{0};
345 insert_block<Real>(z.JI, -J_orig(i.I9, i.I1), row_offset, col_offset);
346 col_offset += i.I1.size();
347 insert_block<Real>(z.JI, -J_orig(i.I9, i.I3), row_offset, col_offset);
348 col_offset += i.I3.size();
349 insert_block<Real>(z.JI, -J_orig(i.I9, i.I4), row_offset, col_offset);
350 col_offset += i.I4.size();
351 insert_block<Real>(z.JI, -J_orig(i.I9, i.I5), row_offset, col_offset);
352 row_offset += i.n9; // next row block
353 col_offset = 0;
354 insert_block<Real>(z.JI, J_orig(i.I9, i.I1), row_offset, col_offset);
355 col_offset += i.I1.size();
356 insert_block<Real>(z.JI, J_orig(i.I9, i.I3), row_offset, col_offset);
357 col_offset += i.I3.size();
358 insert_block<Real>(z.JI, J_orig(i.I9, i.I4), row_offset, col_offset);
359 col_offset += i.I4.size();
360 insert_block<Real>(z.JI, J_orig(i.I9, i.I5), row_offset, col_offset);
361 }
362
363 // Scale objective gradient
364 z.g *= z.fs;
365
366 // Scale constraint Jacobians
367 if (i.nE > 0) {
368 for (Integer k{0}; k < z.JE.outerSize(); ++k) {
369 for (typename SparseMatrix<Real>::InnerIterator it(z.JE, k); it; ++it) {
370 it.valueRef() *= z.cEs[it.row()];
371 }
372 }
373 }
374 if (i.nI > 0) {
375 for (Integer k{0}; k < z.JI.outerSize(); ++k) {
376 for (typename SparseMatrix<Real>::InnerIterator it(z.JI, k); it; ++it) {
377 it.valueRef() *= z.cIs[it.row()];
378 }
379 }
380 }
381 }

395 template <typename Real>
396 inline void evalHessian(Iterate<Real> & z, Input<Real> & i, Counter & c, Problem<Real> const * problem)
397 {
398 // Evaluate lambda in original space
399 Vector<Real> l_orig, x_orig;
400 evalLambdaOriginal<Real>(z, i, l_orig);
401 evalXOriginal<Real>(z, i, x_orig);
402
403 // Initialize/Reset evaluation flag
404 z.err = 0;
405
406 // Increment Hessian evaluation counter
407 incrementHessianCount(c);
408
409 // Try AMPL Hessian evaluation
410 Matrix<Real> H_orig;
411 try
412 {
413 // Evaluate H_orig
414 problem->lagrangian_hessian(x_orig, l_orig, H_orig);
415 }
416 catch (...)
417 {
418 // Set evaluation flag, default values, and return
419 z.err = 1;
420 return;
421 }
422
423 // Set Hessian of the Lagrangian
424 Integer row_offset{0}, col_offset{0};
425 insert_block<Real>(z.H, H_orig(i.I1, i.I1), row_offset, col_offset);
426 col_offset += i.I1.size();
427 insert_block<Real>(z.H, H_orig(i.I1, i.I3), row_offset, col_offset);
428 col_offset += i.I3.size();
429 insert_block<Real>(z.H, H_orig(i.I1, i.I4), row_offset, col_offset);
430 col_offset += i.I4.size();
431 insert_block<Real>(z.H, H_orig(i.I1, i.I5), row_offset, col_offset);
432 row_offset += i.I1.size();
433 col_offset = 0; // next row block
434 insert_block<Real>(z.H, H_orig(i.I3, i.I1), row_offset, col_offset);
435 col_offset += i.I1.size();
436 insert_block<Real>(z.H, H_orig(i.I3, i.I3), row_offset, col_offset);
437 col_offset += i.I3.size();
438 insert_block<Real>(z.H, H_orig(i.I3, i.I4), row_offset, col_offset);
439 col_offset += i.I4.size();
440 insert_block<Real>(z.H, H_orig(i.I3, i.I5), row_offset, col_offset);
441 row_offset += i.I3.size();
442 col_offset = 0; // next row block
443 insert_block<Real>(z.H, H_orig(i.I4, i.I1), row_offset, col_offset);
444 col_offset += i.I1.size();
445 insert_block<Real>(z.H, H_orig(i.I4, i.I3), row_offset, col_offset);
446 col_offset += i.I3.size();
447 insert_block<Real>(z.H, H_orig(i.I4, i.I4), row_offset, col_offset);
448 col_offset += i.I4.size();
449 insert_block<Real>(z.H, H_orig(i.I4, i.I5), row_offset, col_offset);
450 row_offset += i.I4.size();
451 col_offset = 0; // next row block
452 insert_block<Real>(z.H, H_orig(i.I5, i.I1), row_offset, col_offset);
453 col_offset += i.I1.size();
454 insert_block<Real>(z.H, H_orig(i.I5, i.I3), row_offset, col_offset);
455 col_offset += i.I3.size();
456 insert_block<Real>(z.H, H_orig(i.I5, i.I4), row_offset, col_offset);
457 col_offset += i.I4.size();
458 insert_block<Real>(z.H, H_orig(i.I5, i.I5), row_offset, col_offset);
459
460 // Rescale H
461 z.H *= z.rho*z.fs;
462 }

487 template <typename Real>
488 inline Real evalKKTError(Iterate<Real> & z, Input<Real> const & i, Real const rho, Real const mu)
489 {
490 // Initialize optimality vector
491 Vector<Real> kkt(i.nV+2*i.nE+2*i.nI);
492 kkt.setZero();
493
494 // Set gradient of penalty objective
495 kkt.head(i.nV) = rho*z.g;
496
497 // Set gradient of Lagrangian for constraints
498 if (i.nE > 0) {kkt.head(i.nV) += (z.lE.matrix().transpose()*z.JE).transpose();}
499 if (i.nI > 0) {kkt.head(i.nV) += (z.lI.matrix().transpose()*z.JI).transpose();}
500
501 // Set complementarity for constraint slacks
502 if (i.nE > 0) {
503 kkt.segment(i.nV, 2*i.nE) << z.r1*(1.0 + z.lE) - mu, z.r2*(1.0 - z.lE) - mu;
504 }
505 if (i.nI > 0) {
506 kkt.segment(i.nV+2*i.nE, 2*i.nI) << z.s1*z.lI - mu, z.s2*(1.0 - z.lI) - mu;
507 }
508
509 // Scale complementarity
510 if (rho > 0) {
511 kkt *= (1.0 / std::max(1.0, (rho*z.g).template lpNorm<Eigen::Infinity>()));
512 }
513
514 // Evaluate optimality error
515 return kkt.template lpNorm<Eigen::Infinity>();
516 }

540 template <typename Real>
541 inline void evalLambdaOriginal(Iterate<Real> const & z, Input<Real> const & i, Vector<Real> & l)
542 {
543 // Initialize multipliers in original space
544 l.setZero(i.nE+i.n7+i.n8+i.n9);
545
546 // Scale equality constraint multipliers
547 if (i.nE > 0) {l(i.I6) = z.lE*(z.cEs/(z.rho*z.fs));}
548
549 // Scale inequality constraint multipliers
550 Array<Real> lI;
551 if (i.n7+i.n8+i.n9 > 0) {lI = z.lI*(z.cIs/(z.rho*z.fs));}
552
553 // Set inequality constraint multipliers in original space
554 if (i.n7 > 0) {
555 l(i.I7) = -lI.segment(i.n3+i.n4+i.n5+i.n5, i.n7);
556 }
557 if (i.n8 > 0) {
558 l(i.I8) = lI.segment(i.n3+i.n4+i.n5+i.n5+i.n7, i.n8);
559 }
560 if (i.n9 > 0) {
561 l(i.I9) = lI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8+i.n9, i.n9)
562 -lI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8, i.n9);
563 }

590 template <typename Real>
591 inline void evalMerit(Iterate<Real> & z, Input<Real> const & i)
592 {
593 // Initialize merit for objective
594 z.phi = z.rho*z.f;
595
596 // Update merit for slacks
597 if (i.nE > 0) {
598 Array<Real> r_all(2*i.nE); r_all << z.r1, z.r2;
599 z.phi -= z.mu * r_all.log().sum() - r_all.sum();
600 }
601 if (i.nI > 0) {
602 Array<Real> s_all(2*i.nI); s_all << z.s1, z.s2;
603 z.phi -= z.mu * s_all.log().sum() - z.s2.sum();
604 }
605 }

615 template <typename Real>
616 inline void evalNewtonMatrix(Iterate<Real> & z, Parameter<Real> & p, Input<Real> const & i, Counter & c)
617 {
618 // Check for equality constraints
619 if (i.nE > 0)
620 {
621 // Set diagonal terms
622 for (Integer j{0}; j < i.nE; ++j) {
623 z.A.coeffRef(i.nV+j, i.nV+j) = (1.0 + z.lE(j))/z.r1(j);
624 z.A.coeffRef(i.nV+i.nE+j, i.nV+i.nE+j) = (1.0 - z.lE(j))/z.r2(j);
625 }
626
627 // Set constraint Jacobian
628 insert_block<Real>(z.A, z.JE, i.nV+2*i.nE+2*i.nI, 0);
629 }
630
631 // Check for inequality constraints
632 if (i.nI > 0)
633 {
634 // Set diagonal terms
635 const Integer offset{i.nV+2*i.nE};
636 for (Integer j{0}; j < i.nI; ++j) {
637 z.A.coeffRef(offset+j, offset+j) = z.lI(j)/z.s1(j);
638 z.A.coeffRef(offset+i.nI+j, offset+i.nI+j) = (1.0 - z.lI(j))/z.s2(j);
639 }
640
641 // Set constraint Jacobian
642 insert_block<Real>(z.A, z.JI, i.nV+3*i.nE+2*i.nI, 0);
643 }
644
645 // Set minimum potential shift
646 Real min_shift{std::max(p.shift_min, p.shift_factor1*z.shift)};
647
648 // Initialize Hessian modification
649 if (z.cut_ == true) {z.shift = std::min(p.shift_max, min_shift/p.shift_factor2);}
650 else {z.shift = 0;}
651 // Initialize inertia correction loop
652 bool done{false};
653 z.shift22 = 0;
654
655 // Loop until inertia is correct
656 while (!done && z.shift < p.shift_max)
657 {
658 // Set Hessian of Lagrangian
659 insert_block<Real>(z.A, z.H+z.shift*Matrix<Real>::Identity(i.nV, i.nV), 0, 0);
660
661 // Set diagonal terms
662 {
663 const Integer offset{i.nV+2*i.nE+2*i.nI};
664 for (Integer j{0}; j < i.nE; ++j) {
665 z.A.coeffRef(offset+j, offset+j) = -z.shift22;
666 }
667 }
668 {
669 const Integer offset{i.nV+3*i.nE+2*i.nI};
670 // Set diagonal terms
671 for (Integer j{0}; j < i.nI; ++j) {
672 z.A.coeffRef(offset+j, offset+j) = -z.shift22;
673 }
674 }
675
676 // Set number of nonzeros in (upper triangle of) Newton matrix
677 z.Annz = static_cast<Integer>(z.A.nonZeros());
678
679 // Factor primal-dual matrix
680 z.ldlt.compute(z.A);
681
682 // Approximate number of negative pivots (inertia)
683 Integer neig{static_cast<Integer>((z.ldlt.vectorD().array() < 0.0).count())};
684
685 // Increment factorization counter
686 incrementFactorizationCount(c);
687
688 // Set number of nonnegative eigenvalues
689 Integer peig{i.nA - neig};
690
691 // Check inertia
692 if (peig < i.nV+2*i.nE+2*i.nI) {z.shift = std::max(min_shift, z.shift/p.shift_factor2);}
693 else if (neig < i.nE+i.nI && z.shift22 == 0) {z.shift22 = p.shift_min;}
694 else {done = true;}
695 }
696
697 // Update Hessian
698 z.H.diagonal().array() += z.shift;
699
700 // Compress Newton matrix
701 z.A.makeCompressed();
702 }

710 template <typename Real>
711 inline void evalNewtonRhs(Iterate<Real> & z, Input<Real> const & i)
712 {
713 // Initialize right-hand side vector
714 z.b.setZero();
715
716 // Set gradient of objective
717 for (Integer k{0}; k < i.nV; ++k) {
718 for (Integer k{0}; k < i.nV; ++k) {z.b.coeffRef(k) = z.rho*z.g(k);}
719
720 // Set gradient of Lagrangian for constraints
721 if (i.nE > 0) {
722 Vector<Real> tmp((z.lE.matrix().transpose()*z.JE).transpose());
723 for (Integer k{0}; k < tmp.size(); ++k) {z.b.coeffRef(k) += tmp[k];}
724 }
725 if (i.nI > 0) {
726 Vector<Real> tmp((z.lI.matrix().transpose()*z.JI).transpose());
727 for (Integer k{0}; k < tmp.size(); ++k) {z.b.coeffRef(k) += tmp[k];}
728 }
729 }
730
731 // Set complementarity for constraint slacks
732 if (i.nE > 0) {
733 // Compute element-wise complementarity terms with safe element-wise division
734 Vector<Real> tmp(2*i.nE);
735 tmp << 1.0 + z.lE - z.mu * z.r1.cwiseInverse(), 1.0 - z.lE - z.mu * z.r2.cwiseInverse();
736 for (Integer k{0}; k < 2*i.nE; ++k) {z.b.insert(i.nV + k) = tmp[k];}
737 }
738 if (i.nI > 0) {
739 // Compute element-wise complementarity terms with safe element-wise division
740 Vector<Real> tmp(2*i.nI);
741 tmp << z.lI - z.mu * z.s1.cwiseInverse(), 1.0 - z.lI - z.mu * z.s2.cwiseInverse();
742 for (Integer k{0}; k < 2*i.nI; ++k) {z.b.insert(i.nV + 2*i.nE + k) = tmp[k];}
743 }
744
745 // Set penalty-interior-point constraint values
746 if (i.nE > 0) {
747 const Vector<Real> tmp(z.cE + z.r1 - z.r2);
748 const Integer offset{i.nV+2*i.nE+2*i.nI};
749 for (Integer k{0}; k < i.nE; ++k) {z.b.coeffRef(offset+k) = tmp[k];}
750 }
751 if (i.nI > 0) {
752 const Vector<Real> tmp(z.cI + z.s1 - z.s2);
753 const Integer offset{i.nV+3*i.nE+2*i.nI};
754 for (Integer k{0}; k < i.nI; ++k) {z.b.coeffRef(offset+k) = tmp[k];}
755 }
756 }

767 template <typename Real>
768 inline void evalScalings(Iterate<Real> & z, Parameter<Real> & p, Input<Real> & i, Counter & c,
769 Problem<Real> const * problem)
770 {
771 // Initialize scalings
772 z.fs = 1;
773 z.cEs.setOnes(i.nE);
774 z.cIs.setOnes(i.nI);
775
776 // Evaluate gradients
777 evalGradients(z, i, c, problem);
778
779 // Scale down objective if norm of gradient is too large
780 z.fs = p.grad_max / std::max(z.g.template lpNorm<Eigen::Infinity>(), p.grad_max);
781
782 // Loop through equality constraints
783 for (Integer j{0}; j < i.nE; ++j)
784 {
785 // Scale down equality constraint j if norm of gradient is too large
786 Real row_inf_norm{0.0};
787 for (Integer c{0}; c < z.JE.outerSize(); ++c) {
788 for (typename SparseMatrix<Real>::InnerIterator it(z.JE, c); it; ++it) {
789 if (it.row() == j) {row_inf_norm = std::max(row_inf_norm, std::abs(it.value()));}
790 }
791 }
792 z.cEs(j) = p.grad_max / std::max(row_inf_norm, p.grad_max);
793 }
794
795 // Loop through inequality constraints
796 for (Integer j{0}; j < i.nI; ++j)
797 {
798 // Scale down inequality constraint j if norm of gradient is too large
799 Real row_inf_norm{0.0};
800 for (Integer c{0}; c < z.JI.outerSize(); ++c) {
801 for (typename SparseMatrix<Real>::InnerIterator it(z.JI, c); it; ++it) {
802 if (it.row() == j) {row_inf_norm = std::max(row_inf_norm, std::abs(it.value()));}
803 }
804 }
805 z.cIs(j) = p.grad_max / std::max(row_inf_norm, p.grad_max);
806 }

816 template <typename Real>
817 inline void evalSlacks(Iterate<Real> & z, Parameter<Real> & p, Input<Real> const & i)
818 {
819 // Check for equality constraints
820 if (i.nE > 0)
821 {
822 // Set slacks
823 z.r1 = 0.5*((z.mu - z.cE) + (z.cE.square() + z.mu*z.mu).sqrt());
824 z.r2 = 0.5*((z.mu + z.cE) + (z.cE.square() + z.mu*z.mu).sqrt());
825
826 // Adjust for numerical error
827 z.r1.cwiseMax(p.slack_min);
828 z.r2.cwiseMax(p.slack_min);
829 }
830
831 // Check for inequality constraints
832 if (i.nI > 0)
833 {
834 // Set slacks
835 z.s1 = 0.5*((2.0*z.mu - z.cI) + (z.cI.square() + 4.0*z.mu*z.mu).sqrt());
836 z.s2 = 0.5*((2.0*z.mu + z.cI) + (z.cI.square() + 4.0*z.mu*z.mu).sqrt());
837
838 // Adjust for numerical error
839 z.s1.cwiseMax(p.slack_min);
840 z.s2.cwiseMax(p.slack_min);
841 }

888 {

890 z.A.reserve(z.Hnnz + 5*i.nE + 5*i.nI + z.JEnnz + z.JInnz);

1031 template <typename Real>
1032 inline void updateParameters(Iterate<Real> & z, Parameter<Real> & p, Input<Real> & i)
1033 {
1034 // Check for interior-point parameter update based on optimality error
1035 while (z.mu > p.mu_min && z.kkt(2) <= std::max(z.mu, p.opt_err_tol-z.mu))
1036 {
1037 // Restrict interior-point parameter increase
1038 setMuMaxExpZero(p);
1039
1040 // Update interior-point parameter
1041 if (z.mu > p.mu_min)
1042 {
1043 // Decrease interior-point
1044 z.mu = std::max(p.mu_min, std::min(p.mu_factor*z.mu, std::pow(z.mu, p.mu_factor_exp)));
1045
1046 // Evaluate penalty and interior-point parameter dependent quantities
1047 evalDependent(z, p, i);
1048 }
1049 }
1050
1051 // Check for penalty parameter update based on optimality error
1052 if ((z.kkt(1) <= p.opt_err_tol && z.v > p.opt_err_tol) ||
1053 z.v > std::max({1.0, z.v_, p.infeas_max}))
1054 {
1055 // Update penalty parameter
1056 if (z.rho > p.rho_min)
1057 {
1058 // Decrease penalty parameter
1059 z.rho = std::max(p.rho_min, p.rho_factor*z.rho);
1060
1061 // Evaluate penalty and interior-point parameter dependent quantities
1062 evalDependent(z, p, i);
1063 }
1064 }
