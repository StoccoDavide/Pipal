52 template<typename Real>
53 void Solver<Real>::evalLinearCombination(Direction<Real> const & d1, Direction<Real> const & d2,
54 Direction<Real> const & d3, Real const a1, Real const a2, Real const a3)
55 {
56 // Create alias for easier access
57 Input<Real> & i{this->m_input};
58 Direction<Real> & d{this->m_direction};
59
60 // Evaluate linear combinations
61 d.x = a1*d1.x + a2*d2.x + a3*d3.x;
62 if (i.nE > 0) {
63 d.r1 = a1*d1.r1 + a2*d2.r1 + a3*d3.r1;
64 d.r2 = a1*d1.r2 + a2*d2.r2 + a3*d3.r2;
65 d.lE = a1*d1.lE + a2*d2.lE + a3*d3.lE;
66 }
67 if (i.nI > 0) {
68 d.s1 = a1*d1.s1 + a2*d2.s1 + a3*d3.s1;
69 d.s2 = a1*d1.s2 + a2*d2.s2 + a3*d3.s2;
70 d.lI = a1*d1.lI + a2*d2.lI + a3*d3.lI;
71 }
72
73 // Evaluate primal direction norm
74 d.x_norm = d.x.norm();
75
76 // Evaluate dual direction norm
77 d.l_norm = std::sqrt(d.lE.matrix().squaredNorm() + d.lI.matrix().squaredNorm());
78 }

84 template<typename Real>
85 void Solver<Real>::evalModels()
86 {
87 // Create alias for easier access
88 Input<Real> & i{this->m_input};
89 Iterate<Real> & z{this->m_iterate};
90 Direction<Real> & d{this->m_direction};
91
92 // Evaluate reduction in linear model of penalty-interior-point objective for zero penalty parameter
93 d.lred0 = 0;
94 if (i.nE > 0) {
95 d.lred0 -= ((1.0 - z.mu / z.r1) * d.r1).sum();
96 d.lred0 -= ((1.0 - z.mu / z.r2) * d.r2).sum();
97 }
98 if (i.nI > 0) {
99 d.lred0 -= (((-z.mu) / z.s1) * d.s1).sum();
100 d.lred0 -= ((1.0 - z.mu / z.s2) * d.s2).sum();
101 }
102
103 // Evaluate remaining quantities only for nonzero penalty parameter
104 if (z.rho > 0)
105 {
106 // Evaluate reduction in linear model of merit function for zero penalty parameter
107 d.ltred0 = 0;
108 if (i.nE > 0) {
109 Array<Real> sqrt_term((z.cE.square() + z.mu*z.mu).sqrt());
110 d.ltred0 -= 0.5*(((1.0 - z.mu/z.r1) * (-1.0 + z.cE/sqrt_term) + (1.0 - z.mu/z.r2) *
111 (1.0 + z.cE/sqrt_term)) * (z.JE * d.x).array()).sum();
112 }
113 if (i.nI > 0) {
114 Array<Real> sqrt_term((z.cI.square() + 4.0*z.mu*z.mu).sqrt());
115 d.ltred0 -= 0.5*((((-z.mu)/z.s1) * (-1.0 + z.cI/sqrt_term) + (1.0 - z.mu/z.s2) *
116 (1.0 + z.cI/sqrt_term)) * (z.JI * d.x).array()).sum();
117 }
118
119 // Evaluate reduction in linear model of merit function
120 d.ltred = -z.rho*z.g.transpose()*d.x + d.ltred0;
121
122 // Evaluate reduction in quadratic model of merit function
123 d.qtred = d.ltred - 0.5*d.x.transpose()*z.H*d.x;
124 if (i.nE > 0) {
125 Array<Real> Jd(z.JE*d.x);
126 Array<Real> Dinv((z.r1/(1.0+z.lE) + z.r2/(1.0-z.lE)).matrix());
127 d.qtred -= 0.5*Jd.matrix().transpose() * ((Jd/Dinv).matrix());
128 }
129 if (i.nI > 0) {
130 Array<Real> Jd(z.JI*d.x);
131 Array<Real> Dinv((z.s1/(0.0+z.lI) + z.s2/(1.0-z.lI)).matrix());
132 d.qtred -= 0.5*Jd.matrix().transpose() * ((Jd/Dinv).matrix());
133 }
134
135 // Initialize quality function vector
136 Array<Real> vec(i.nV+2*i.nE+2*i.nI);
137 vec.setZero();
138
139 // Set gradient of objective
140 vec.head(i.nV) = z.rho*z.g;
141
142 // Set gradient of Lagrangian for constraints
143 if (i.nE > 0) {vec.head(i.nV) += ((z.lE+d.lE).matrix().transpose()*z.JE).transpose().array();}
144 if (i.nI > 0) {vec.head(i.nV) += ((z.lI+d.lI).matrix().transpose()*z.JI).transpose().array();}
145
146 // Set complementarity for constraint slacks
147 if (i.nE > 0) {
148 vec.segment(i.nV, 2*i.nE) <<
149 (z.r1+d.r1)*(1.0 + (z.lE+d.lE)), (z.r2+d.r2)*(1.0 - (z.lE+d.lE));
150 }
151 if (i.nI > 0) {
152 vec.segment(i.nV+2*i.nE, 2*i.nI) <<
153 (z.s1+d.s1)*(z.lI+d.lI), (z.s2+d.s2)*(1.0 - (z.lI+d.lI));
154 }
155
156 // Evaluate quality function
157 d.m = vec.matrix().template lpNorm<Eigen::Infinity>();
158 }

165 template<typename Real>
166 void Solver<Real>::evalNewtonStep()
167 {
168 // Create alias for easier access
169 Input<Real> const & i{this->m_input};
170 Iterate<Real> const & z{this->m_iterate};
171 Direction<Real> & d{this->m_direction};
172
173 // Evaluate direction
174 Vector<Real> dir(z.ldlt.solve(-z.b));
175
176 // Parse direction
177 d.x = dir.head(i.nV);
178 if (i.nE > 0) {
179 d.r1 = dir.segment(i.nV, i.nE);
180 d.r2 = dir.segment(i.nV+i.nE, i.nE);
181 d.lE = dir.segment(i.nV+i.nE+i.nE+i.nI+i.nI, i.nE);
182 }
183 if (i.nI > 0) {
184 d.s1 = dir.segment(i.nV+i.nE+i.nE, i.nI);
185 d.s2 = dir.segment(i.nV+i.nE+i.nE+i.nI, i.nI);
186 d.lI = dir.segment(i.nV+i.nE+i.nE+i.nI+i.nI+i.nE, i.nI);
187 }
188
189 // Evaluate primal direction norm
190 d.x_norm = d.x.norm();
191
192 // Evaluate dual direction norm
193 d.l_norm = std::sqrt(d.lE.matrix().squaredNorm() + d.lI.matrix().squaredNorm());
194 }

200 template <typename Real>
201 void Solver<Real>::evalStep()
202 {
203 // Create alias for easier access
204 Parameter<Real> & p{this->m_parameter};
205 Iterate<Real> & z{this->m_iterate};
206 Direction<Real> & d{this->m_direction};
207
208 // Reset maximum exponent for interior-point parameter increases
209 this->resetMuMaxExp();
210
211 // Update penalty-interior-point parameters based on KKT errors
212 this->updateParameters();
213
214 // Evaluate matrices
215 this->evalMatrices();
216
217 // Set last penalty parameter
218 this->setRhoLast(z.rho);
219
220 // Check for aggressive algorithm
221 if (p.algorithm == Algorithm::ADAPTIVE)
222 {
223 // Check KKT memory for potential mu increase limit
224 if (z.kkt(1) > z.kkt_.maxCoeff()) {this->setMuMaxExpZero();}
225
226 // Store current penalty and interior-point parameters
227 Real rho_curr{z.rho}, mu_curr{z.mu};
228
229 // Evaluate trial steps
230 Direction<Real> d1, d2, d3;
231 this->resetDirection(d1);
232 this->resetDirection(d2);
233 this->resetDirection(d3);
234 this->evalTrialSteps(d1, d2, d3);
235
236 // Set trial interior-point parameter values
237 Array<Real> exponents(Array<Real>::LinSpaced(p.mu_trials, p.mu_trials - 1, 0) - p.mu_max_exp);
238 Array<Real> Mu((mu_curr * exponents.unaryExpr(
239 [&p] (Real e) {return std::pow(p.mu_factor, e);})
240 ).min(p.mu_max).max(p.mu_min));
241
242 // Initialize feasibility direction data
243 Vector<Real> lred0_0_mu(p.mu_trials);
244 lred0_0_mu.setZero();
245
246 // Loop through interior-point parameter values
247 for (Integer j{0}; j < p.mu_trials; ++j)
248 {
249 // Set penalty and interior-point parameters
250 this->setRho(0.0);
251 this->setMu(Mu(j));
252
253 // Evaluate direction
254 this->evalLinearCombination(d1, d2, d3,
255 z.rho/rho_curr+z.mu/mu_curr-1.0, 1.0-z.mu/mu_curr, 1.0-z.rho/rho_curr);
256
257 // Cut length
258 d.x *= std::min(d.x_norm_/std::max(d.x_norm, 1.0), 1.0);
259
260 // Run fraction-to-boundary
261 this->fractionToBoundary();
262
263 // Cut length
264 this->evalTrialStepCut();
265
266 // Evaluate models
267 this->evalModels();
268
269 // Set feasibility direction data
270 lred0_0_mu(j) = d.lred0;
271 }
272
273 // Initialize updating data
274 Vector<Real> ltred0_rho_mu(p.mu_trials), qtred_rho_mu(p.mu_trials), m_rho_mu(p.mu_trials);
275 ltred0_rho_mu.setZero(); qtred_rho_mu.setZero(); m_rho_mu.setZero();
276
277 // Initialize check
278 bool check{false};
279
280 // Loop through penalty parameter values
281 for (Integer k{0}; k < p.rho_trials; ++k)
282 {
283 // Set penalty parameter
284 this->setRho(std::max(p.rho_min, std::pow(p.rho_factor, k)*rho_curr) );
285
286 // Set last penalty parameter
287 if (rho_curr > z.kkt(0)*z.kkt(0)) {this->setRhoLast(z.rho );}
288
289 // Loop through interior-point parameter values
290 for (Integer j{0}; j < p.mu_trials; ++j)
291 {
292 // Set interior-point parameter
293 this->setMu(Mu(j));
294
295 // Evaluate direction
296 this->evalLinearCombination(
297 d1, d2, d3,
298 z.rho/rho_curr+z.mu/mu_curr-1.0,
299 1.0-z.mu/mu_curr,
300 1.0-z.rho/rho_curr
301 );
302
303 // Run fraction-to-boundary
304 this->fractionToBoundary();
305
306 // Cut steps
307 this->evalTrialStepCut();
308
309 // Evaluate models
310 this->evalModels();
311
312 // Set updating data
313 ltred0_rho_mu(j) = d.ltred0;
314 qtred_rho_mu(j) = d.qtred;
315 m_rho_mu(j) = d.m;
316
317 // Check updating conditions for infeasible points
318 if (z.v > p.opt_err_tol && (ltred0_rho_mu(j) < p.update_con_1*lred0_0_mu(j) ||
319 qtred_rho_mu(j) < p.update_con_2*lred0_0_mu(j) || z.rho > z.kkt(0)*z.kkt(0))) {
320 m_rho_mu(j) = std::numeric_limits<Real>::infinity();
321 }
322
323 // Check updating conditions for feasible points
324 if (z.v <= p.opt_err_tol && qtred_rho_mu(j) < 0.0) {
325 m_rho_mu(j) = std::numeric_limits<Real>::infinity();
326 }
327 }
328
329 // Find minimum m for current rho
330 Real m_min{m_rho_mu.minCoeff()};
331
332 // Check for finite minimum
333 if (m_min < std::numeric_limits<Real>::infinity())
334 {
335 // Loop through mu values
336 for (Integer j{0}; j < p.mu_trials; ++j)
337 {
338 // Check condition
339 if (m_rho_mu(j) <= p.update_con_3*m_min) {this->setMu(Mu(j));}
340 }
341
342 // Set condition check
343 check = true;
344
345 // Break loop
346 break;
347 }
348 }
349
350 // Check conditions
351 if (check == false) {this->setRho(rho_curr); this->setMu(mu_curr);}
352
353 // Evaluate merit
354 this->evalMerit();
355 }
356
357 // Evaluate primal-dual right-hand side vector
358 this->evalNewtonRhs();
359
360 // Evaluate search direction
361 this->evalNewtonStep();
362
363 // Evaluate models
364 this->evalModels();
365
366 // Store last direction norm
367 d.x_norm_ = d.x_norm;
368 }

411 template<typename Real>
412 void Solver<Real>::evalTrialSteps(Direction<Real> & d1, Direction<Real> & d2, Direction<Real> & d3)
413 {
414 // Create alias for easier access
415 Iterate<Real> const & z{this->m_iterate};
416
417 // Store current penalty and interior-point parameters
418 Real rho_curr{z.rho}, mu_curr{z.mu};
419
420 // Evaluate direction for current penalty and interior-point parameters
421 this->setRho(rho_curr);
422 this->setMu(mu_curr);
423 this->evalNewtonRhs();
424 this->evalNewtonStep();
425 this->evalTrialStep(d1);
426
427 // Evaluate direction for zero interior-point parameter
428 this->setRho(rho_curr);
429 this->setMu(0.0);
430 this->evalNewtonRhs();
431 this->evalNewtonStep();
432 this->evalTrialStep(d2);
433
434 // Evaluate direction for zero penalty parameter
435 this->setRho(0.0);
436 this->setMu(mu_curr);
437 this->evalNewtonRhs();
438 this->evalNewtonStep();
439 this->evalTrialStep(d3);
440 }
