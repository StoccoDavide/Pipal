22 template <typename Real>
23 void Solver<Real>::buildIterate()
24 {
25 // Create alias for easier access
26 Parameter<Real> & p{this->m_parameter};
27 Input<Real> & i{this->m_input};
28 Iterate<Real> & z{this->m_iterate};
29
30 // Safe initialization of all members
31 z.f = 0;
32 z.fu = 0;
33 z.g.setZero(i.nV);
34 z.r1.setZero(i.nE);
35 z.r2.setZero(i.nE);
36 z.cE.setZero(i.nE);
37 z.JE.resize(i.nE, i.nV);
38 z.s1.setZero(i.nI);
39 z.s2.setZero(i.nI);
40 z.cI.setZero(i.nI);
41 z.JI.resize(i.nI, i.nV);
42 z.H.resize(i.nV, i.nV);
43 z.v = 0;
44 z.vu = 0;
45 z.phi = 0;
46 z.Annz = 0;
47 z.shift = 0;
48 z.b.resize(i.nA);
49 z.kkt.setZero(3);
50 z.kkt_.setConstant(p.opt_err_mem, std::numeric_limits<Real>::infinity());
51 z.fs = 1.0;
52 z.cEs.setOnes(i.nE);
53 z.cEu.setZero(i.nE);
54 z.cIs.setOnes(i.nI);
55 z.cIu.setZero(i.nI);
56 z.A.resize(i.nA, i.nA);
57 z.shift22 = 0;
58 z.cut_ = false;
59
60 // Initialize point
61 z.x = i.x0;
62 z.rho = p.rho_init;
63 z.mu = p.mu_init;
64 z.lE.setZero(i.nE);
65 z.lI.setConstant(i.nI, 0.5);
66 z.err = 0;
67 this->evalScalings();
68 this->evalFunctions();
69 this->evalGradients();
70 this->evalDependent();
71 z.v0 = 1.0;
72 this->evalInfeasibility(z);
73 z.v0 = z.v;
74 this->evalInfeasibility(z);
75 z.v_ = z.v;
76 this->evalHessian();
77 z.Hnnz = static_cast<Integer>(z.H.nonZeros());
78 z.JEnnz = static_cast<Integer>(z.JE.nonZeros());
79 z.JInnz = static_cast<Integer>(z.JI.nonZeros());
80 initNewtonMatrix();
81 this->evalNewtonMatrix();
82 }

92 template <typename Real>
93 Integer Solver<Real>::checkTermination() const
94 {
95 // Create alias for easier access
96 Parameter<Real> const & p{this->m_parameter};
97 Counter const & c{this->m_counter};
98 Input<Real> const & i{this->m_input};
99 Iterate<Real> const & z{this->m_iterate};
100
101 // Update termination based on optimality error of nonlinear optimization problem
102 if (z.kkt(1) <= p.opt_err_tol && z.v <= p.opt_err_tol) return 1;
103
104 // Update termination based on optimality error of feasibility problem
105 if (z.kkt(0) <= p.opt_err_tol && z.v > p.opt_err_tol) return 2;
106
107 // Update termination based on iteration count
108 if (c.k >= p.iter_max) return 3;
109
110 // Update termination based on invalid bounds
111 if (i.vi > 0) return 4;
112
113 // Update termination based on function evaluation error
114 if (z.err > 0) return 5;
115
116 return 0;
117 }

123 template <typename Real>
124 void Solver<Real>::evalFunctions()
125 {
126 // Create alias for easier access
127 Input<Real> & i{this->m_input};
128 Iterate<Real> & z{this->m_iterate};
129
130 // Evaluate x in original space
131 Vector<Real> x_orig;
132 this->evalXOriginal(x_orig);
133
134 // Initialize/Reset evaluation flag
135 z.err = 0;
136
137 // Increment function evaluation counter
138 incrementFunctionCount();
139
140 // Try AMPL functions evaluation
141 Vector<Real> c_orig;
142 try
143 {
144 // Evaluate AMPL functions
145 m_problem->objective(x_orig, z.f);
146 m_problem->constraints(x_orig, c_orig);
147 }
148 catch (...)
149 {
150 // Set evaluation flag, default values, and return
151 z.err = 1;
152 z.f = std::numeric_limits<Real>::quiet_NaN();
153 z.cE.setConstant(i.nE, std::numeric_limits<Real>::quiet_NaN());
154 z.cI.setConstant(i.nI, std::numeric_limits<Real>::quiet_NaN());
155 z.fu = std::numeric_limits<Real>::quiet_NaN();
156 z.cEu.setConstant(i.nE, std::numeric_limits<Real>::quiet_NaN());
157 z.cIu.setConstant(i.nI, std::numeric_limits<Real>::quiet_NaN());
158 return;
159 }
160
161 // Set equality constraint values
162 if (i.nE > 0) {z.cE = c_orig(i.I6) - i.b6;}
163
164 // Initialize inequality constraint values
165 if (i.nI > 0) {z.cI.setZero(i.nI);}
166
167 // Set inequality constraint values
168 if (i.n3 > 0) {
169 z.cI.head(i.n3) = i.l3 - z.x.segment(i.n1, i.n3);
170 }
171 if (i.n4 > 0) {
172 z.cI.segment(i.n3, i.n4) = -i.u4 + z.x.segment(i.n1+i.n3, i.n4);
173 }
174 if (i.n5 > 0) {
175 z.cI.segment(i.n3+i.n4, i.n5) = i.l5 - z.x.segment(i.n1+i.n3+i.n4, i.n5);
176 z.cI.segment(i.n3+i.n4+i.n5, i.n5) = -i.u5 + z.x.segment(i.n1+i.n3+i.n4, i.n5);
177 }
178 if (i.n7 > 0) {
179 z.cI.segment(i.n3+i.n4+i.n5+i.n5, i.n7) = i.l7 - c_orig(i.I7);
180 }
181 if (i.n8 > 0) {
182 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7, i.n8) = -i.u8 + c_orig(i.I8);
183 }
184 if (i.n9 > 0) {
185 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8, i.n9) = i.l9 - c_orig(i.I9);
186 z.cI.segment(i.n3+i.n4+i.n5+i.n5+i.n7+i.n8+i.n9, i.n9) = -i.u9 + c_orig(i.I9);
187 }
188
189 // Store unscaled quantities
190 z.fu = z.f;
191 if (i.nE > 0) {z.cEu = z.cE;}
192 if (i.nI > 0) {z.cIu = z.cI;}
193
194 // Scale quantities
195 z.f *= z.fs;
196 if (i.nE > 0) {z.cE = (z.cEs*z.cE).matrix();}
197 if (i.nI > 0) {z.cI = (z.cIs*z.cI).matrix();}
198
199 }

211 template <typename Real>
212 void Solver<Real>::evalGradients()
213 {
214 // Create alias for easier access
215 Input<Real> & i{this->m_input};
216 Iterate<Real> & z{this->m_iterate};
217
218 // Evaluate x in original space
219 Vector<Real> x_orig;
220 this->evalXOriginal(x_orig);
221
222 // Initialize/Reset evaluation flag
223 z.err = 0;
224
225 // Increment gradient evaluation counter
226 incrementGradientCount();
227
228 // Try AMPL gradients evaluation
229 Vector<Real> g_orig;
230 SparseMatrix<Real> J_orig;
231 try
232 {
233 // Evaluate AMPL gradients
234 m_problem->objective_gradient(x_orig, g_orig);
235 m_problem->constraints_jacobian(x_orig, J_orig);
236 }
237 catch (...)
238 {
239 // Set evaluation flag, default values, and return
240 z.err = 1;
241 return;
242 }
243
244 // Set objective gradient
245 z.g << g_orig(i.I1), g_orig(i.I3), g_orig(i.I4), g_orig(i.I5);
246
247 // Initialize equality constraint Jacobian
248 z.JE.setZero();
249
250 // Set equality constraint Jacobian
251 if (i.nE > 0) {
252 Integer col_offset{0};
253 Pipal::insert_block<Real>(z.JE, J_orig, i.I6, i.I1, 0, col_offset);
254 col_offset += i.I1.size();
255 Pipal::insert_block<Real>(z.JE, J_orig, i.I6, i.I3, 0, col_offset);
256 col_offset += i.I3.size();
257 Pipal::insert_block<Real>(z.JE, J_orig, i.I6, i.I4, 0, col_offset);
258 col_offset += i.I4.size();
259 Pipal::insert_block<Real>(z.JE, J_orig, i.I6, i.I5, 0, col_offset);
260 }
261
262 // Initialize inequality constraint Jacobian
263 z.JI.setZero();
264
265 // Set inequality constraint Jacobian
266 if (i.n3 > 0) {
267 for (Integer k{0}; k < i.n3; ++k) {z.JI.coeffRef(k, i.n1+k) = -1.0;}
268 }
269 if (i.n4 > 0) {
270 Integer tmp{i.n1+i.n3};
271 for (Integer k{0}; k < i.n4; ++k) {z.JI.coeffRef(i.n3+k, tmp+k) = 1.0;}
272 }
273 if (i.n5 > 0) {
274 Integer tmp_row{i.n3+i.n4}, tmp_col{i.n1+i.n3+i.n4};
275 for (Integer k{0}; k < i.n5; ++k) {
276 z.JI.coeffRef(tmp_row+k, tmp_col+k) = -1.0;
277 z.JI.coeffRef(tmp_row+i.n5+k, tmp_col+k) = 1.0;
278 }
279 }
280 if (i.n7 > 0) {
281 Integer row_offset{i.n3+i.n4+i.n5+i.n5}, col_offset{0};
282 Pipal::insert_block<Real>(z.JI, -J_orig, i.I7, i.I1, row_offset, col_offset);
283 col_offset += i.I1.size();
284 Pipal::insert_block<Real>(z.JI, J_orig, i.I7, i.I3, row_offset, col_offset);
285 col_offset += i.I3.size();
286 Pipal::insert_block<Real>(z.JI, J_orig, i.I7, i.I4, row_offset, col_offset);
287 col_offset += i.I4.size();
288 Pipal::insert_block<Real>(z.JI, J_orig, i.I7, i.I5, row_offset, col_offset);
289 }
290 if (i.n8 > 0) {
291 Integer row_offset{i.n3+i.n4+i.n5+i.n5+i.n7}, col_offset{0};
292 Pipal::insert_block<Real>(z.JI, J_orig, i.I8, i.I1, row_offset, col_offset);
293 col_offset += i.I1.size();
294 Pipal::insert_block<Real>(z.JI, J_orig, i.I8, i.I3, row_offset, col_offset);
295 col_offset += i.I3.size();
296 Pipal::insert_block<Real>(z.JI, J_orig, i.I8, i.I4, row_offset, col_offset);
297 col_offset += i.I4.size();
298 Pipal::insert_block<Real>(z.JI, J_orig, i.I8, i.I5, row_offset, col_offset);
299 }
300 if (i.n9 > 0) {
301 Integer row_offset{i.n3+i.n4+i.n5+i.n5+i.n7+i.n8}, col_offset{0};
302 Pipal::insert_block<Real>(z.JI, -J_orig, i.I9, i.I1, row_offset, col_offset);
303 col_offset += i.I1.size();
304 Pipal::insert_block<Real>(z.JI, -J_orig, i.I9, i.I3, row_offset, col_offset);
305 col_offset += i.I3.size();
306 Pipal::insert_block<Real>(z.JI, -J_orig, i.I9, i.I4, row_offset, col_offset);
307 col_offset += i.I4.size();
308 Pipal::insert_block<Real>(z.JI, -J_orig, i.I9, i.I5, row_offset, col_offset);
309 row_offset += i.n9; // Next row block
310 col_offset = 0;
311 Pipal::insert_block<Real>(z.JI, J_orig, i.I9, i.I1, row_offset, col_offset);
312 col_offset += i.I1.size();
313 Pipal::insert_block<Real>(z.JI, J_orig, i.I9, i.I3, row_offset, col_offset);
314 col_offset += i.I3.size();
315 Pipal::insert_block<Real>(z.JI, J_orig, i.I9, i.I4, row_offset, col_offset);
316 col_offset += i.I4.size();
317 Pipal::insert_block<Real>(z.JI, J_orig, i.I9, i.I5, row_offset, col_offset);
318 }
319
320 // Scale objective gradient
321 z.g *= z.fs;
322
323 // Scale constraint Jacobians
324 if (i.nE > 0) {
325 for (Integer k{0}; k < z.JE.outerSize(); ++k) {
326 for (typename SparseMatrix<Real>::InnerIterator it(z.JE, k); it; ++it) {
327 it.valueRef() *= z.cEs[it.row()];
328 }
329 }
330 }
331 if (i.nI > 0) {
332 for (Integer k{0}; k < z.JI.outerSize(); ++k) {
333 for (typename SparseMatrix<Real>::InnerIterator it(z.JI, k); it; ++it) {
334 it.valueRef() *= z.cIs[it.row()];
335 }
336 }
337 }

353 Iterate<Real> & z{this->m_iterate};
354

362
363 // Increment Hessian evaluation counter
364 incrementHessianCount();
365
366 // Try AMPL Hessian evaluation

389 row_offset += i.I1.size();
390 col_offset = 0; // Next row block
391 Pipal::insert_block<Real>(z.H, H_orig, i.I3, i.I1, row_offset, col_offset);
392 col_offset += i.I1.size();
393 Pipal::insert_block<Real>(z.H, H_orig, i.I3, i.I3, row_offset, col_offset);
394 col_offset += i.I3.size();
395 Pipal::insert_block<Real>(z.H, H_orig, i.I3, i.I4, row_offset, col_offset);
396 col_offset += i.I4.size();
397 Pipal::insert_block<Real>(z.H, H_orig, i.I3, i.I5, row_offset, col_offset);
398 row_offset += i.I3.size();
399 col_offset = 0; // Next row block
400 Pipal::insert_block<Real>(z.H, H_orig, i.I4, i.I1, row_offset, col_offset);
401 col_offset += i.I1.size();
402 Pipal::insert_block<Real>(z.H, H_orig, i.I4, i.I3, row_offset, col_offset);

404 Pipal::insert_block<Real>(z.H, H_orig, i.I4, i.I4, row_offset, col_offset);

406 Pipal::insert_block<Real>(z.H, H_orig, i.I4, i.I5, row_offset, col_offset);

408 col_offset = 0; // Next row block

421 z.H *= z.rho*z.fs;

431 template <typename Real>
432 Real Solver<Real>::evalKKTError(Real const rho, Real const mu)
433 {
434 // Create alias for easier access
435 Input<Real> & i{this->m_input};
436 Iterate<Real> & z{this->m_iterate};
437
438 // Initialize optimality vector
439 Vector<Real> kkt(i.nV+2*i.nE+2*i.nI);
440 kkt.setZero();
441
442 // Set gradient of penalty objective
443 kkt.head(i.nV) = rho*z.g;
444
445 // Set gradient of Lagrangian for constraints
446 if (i.nE > 0) {kkt.head(i.nV) += (z.lE.matrix().transpose()*z.JE).transpose();}
447 if (i.nI > 0) {kkt.head(i.nV) += (z.lI.matrix().transpose()*z.JI).transpose();}
448
449 // Set complementarity for constraint slacks
450 if (i.nE > 0) {
451 kkt.segment(i.nV, 2*i.nE) << z.r1*(1.0 + z.lE) - mu, z.r2*(1.0 - z.lE) - mu;
452 }
453 if (i.nI > 0) {
454 kkt.segment(i.nV+2*i.nE, 2*i.nI) << z.s1*z.lI - mu, z.s2*(1.0 - z.lI) - mu;
455 }
456
457 // Scale complementarity
458 if (rho > 0) {
459 kkt *= (1.0 / std::max(1.0, (rho*z.g).template lpNorm<Eigen::Infinity>()));
460 }
461
462 // Evaluate optimality error
463 return kkt.template lpNorm<Eigen::Infinity>();
464 }

547 template <typename Real>
548 void Solver<Real>::evalNewtonMatrix()
549 {
550 // Create alias for easier access
551 Parameter<Real> & p{this->m_parameter};
552 Input<Real> & i{this->m_input};
553 Iterate<Real> & z{this->m_iterate};
554
555 // Check for equality constraints
556 if (i.nE > 0)
557 {
558 // Set diagonal terms
559 for (Integer j{0}; j < i.nE; ++j) {
560 z.A.coeffRef(i.nV+j, i.nV+j) = (1.0 + z.lE(j))/z.r1(j);
561 z.A.coeffRef(i.nV+i.nE+j, i.nV+i.nE+j) = (1.0 - z.lE(j))/z.r2(j);
562 }
563
564 // Set constraint Jacobian
565 Pipal::insert_block<Real>(z.A, z.JE, i.nV+2*i.nE+2*i.nI, 0);
566 }
567
568 // Check for inequality constraints
569 if (i.nI > 0)
570 {
571 // Set diagonal terms
572 const Integer offset{i.nV+2*i.nE};
573 for (Integer j{0}; j < i.nI; ++j) {
574 z.A.coeffRef(offset+j, offset+j) = z.lI(j)/z.s1(j);
575 z.A.coeffRef(offset+i.nI+j, offset+i.nI+j) = (1.0 - z.lI(j))/z.s2(j);
576 }
577
578 // Set constraint Jacobian
579 Pipal::insert_block<Real>(z.A, z.JI, i.nV+3*i.nE+2*i.nI, 0);
580 }
581
582 // Set minimum potential shift
583 Real min_shift{std::max(p.shift_min, p.shift_factor1*z.shift)};
584
585 // Initialize Hessian modification
586 if (z.cut_ == true) {z.shift = std::min(p.shift_max, min_shift/p.shift_factor2);}
587 else {z.shift = 0;}
588 // Initialize inertia correction loop
589 bool done{false};
590 z.shift22 = 0;
591
592 // Loop until inertia is correct
593 while (!done && z.shift < p.shift_max)
594 {
595 // Set Hessian of Lagrangian
596 Pipal::insert_block<Real>(z.A, z.H+z.shift*Matrix<Real>::Identity(i.nV, i.nV), 0, 0);
597
598 // Set diagonal terms
599 {
600 const Integer offset{i.nV+2*i.nE+2*i.nI};
601 for (Integer j{0}; j < i.nE; ++j) {
602 z.A.coeffRef(offset+j, offset+j) = -z.shift22;
603 }
604 }
605 {
606 const Integer offset{i.nV+3*i.nE+2*i.nI};
607 // Set diagonal terms
608 for (Integer j{0}; j < i.nI; ++j) {
609 z.A.coeffRef(offset+j, offset+j) = -z.shift22;
610 }
611 }
612
613 // Set number of nonzeros in (upper triangle of) Newton matrix
614 z.Annz = static_cast<Integer>(z.A.nonZeros());
615
616 // Factor primal-dual matrix
617 z.ldlt.compute(z.A);
618
619 // Approximate number of negative pivots (inertia)
620 Integer neig{static_cast<Integer>((z.ldlt.vectorD().array() < 0.0).count())};
621
622 // Increment factorization counter
623 incrementFactorizationCount();
624
625 // Set number of nonnegative eigenvalues
626 Integer peig{i.nA - neig};
627
628 // Check inertia
629 if (peig < i.nV+2*i.nE+2*i.nI) {z.shift = std::max(min_shift, z.shift/p.shift_factor2);}
630 else if (neig < i.nE+i.nI && z.shift22 == 0) {z.shift22 = p.shift_min;}
631 else {done = true;}
632 }
633
634 // Update Hessian
635 z.H.diagonal().array() += z.shift;
636
637 // Compress Newton matrix
638 z.A.makeCompressed();
639 }

646 void Solver<Real>::evalNewtonRhs()
647 {
648 // Create alias for easier access
649 Input<Real> & i{this->m_input};
650 Iterate<Real> & z{this->m_iterate};
651
652 // Initialize right-hand side
653 z.b.setZero();
654
655 // Set gradient of objective
656 z.b.head(i.nV) = z.rho*z.g;
657
658 // Set gradient of Lagrangian for constraints
659 if (i.nE > 0) {z.b.head(i.nV) += (z.lE.matrix().transpose()*z.JE).transpose();}
660 if (i.nI > 0) {z.b.head(i.nV) += (z.lI.matrix().transpose()*z.JI).transpose();}
661
662 // Set complementarity for constraint slacks
663 if (i.nE > 0) {
664 // Compute element-wise complementarity terms with safe element-wise division
665 Vector<Real> tmp(2*i.nE);
666 tmp << 1.0 + z.lE - z.mu * z.r1.cwiseInverse(), 1.0 - z.lE - z.mu * z.r2.cwiseInverse();
667 z.b.segment(i.nV, 2*i.nE) = tmp;
668 }
669 if (i.nI > 0) {
670 // Compute element-wise complementarity terms with safe element-wise division
671 Vector<Real> tmp(2*i.nI);
672 tmp << z.lI - z.mu * z.s1.cwiseInverse(), 1.0 - z.lI - z.mu * z.s2.cwiseInverse();
673 z.b.segment(i.nV + 2*i.nE, 2*i.nI) = tmp;
674 }
675
676 // Set penalty-interior-point constraint values
677 if (i.nE > 0) {z.b.segment(i.nV+2*i.nE+2*i.nI, i.nE) = z.cE + z.r1 - z.r2;}
678 if (i.nI > 0) {z.b.segment(i.nV+3*i.nE+2*i.nI, i.nI) = z.cI + z.s1 - z.s2;}

685 template <typename Real>
686 void Solver<Real>::evalScalings()
687 {
688 // Create alias for easier access
689 Parameter<Real> & p{this->m_parameter};
690 Input<Real> & i{this->m_input};
691 Iterate<Real> & z{this->m_iterate};
692
693 // Initialize scalings
694 z.fs = 1;
695 z.cEs.setOnes(i.nE);
696 z.cIs.setOnes(i.nI);
697
698 // Evaluate gradients
699 this->evalGradients();
700
701 // Scale down objective if norm of gradient is too large
702 z.fs = p.grad_max / std::max(z.g.template lpNorm<Eigen::Infinity>(), p.grad_max);
703
704 // Loop through equality constraints
705 for (Integer j{0}; j < i.nE; ++j)
706 {
707 // Scale down equality constraint j if norm of gradient is too large
708 Real row_inf_norm{0.0};
709 for (Integer c{0}; c < z.JE.outerSize(); ++c) {
710 for (typename SparseMatrix<Real>::InnerIterator it(z.JE, c); it; ++it) {
711 if (it.row() == j) {row_inf_norm = std::max(row_inf_norm, std::abs(it.value()));}
712 }
713 }
714 z.cEs(j) = p.grad_max / std::max(row_inf_norm, p.grad_max);
715 }
716
717 // Loop through inequality constraints
718 for (Integer j{0}; j < i.nI; ++j)
719 {
720 // Scale down inequality constraint j if norm of gradient is too large
721 Real row_inf_norm{0.0};
722 for (Integer c{0}; c < z.JI.outerSize(); ++c) {
723 for (typename SparseMatrix<Real>::InnerIterator it(z.JI, c); it; ++it) {
724 if (it.row() == j) {row_inf_norm = std::max(row_inf_norm, std::abs(it.value()));}
725 }
726 }
727 z.cIs(j) = p.grad_max / std::max(row_inf_norm, p.grad_max);
728 }

736 void Solver<Real>::evalSlacks()
737 {
738 // Create alias for easier access
739 Parameter<Real> & p{this->m_parameter};
740 Input<Real> & i{this->m_input};
741 Iterate<Real> & z{this->m_iterate};
742
743 // Check for equality constraints
744 if (i.nE > 0)
745 {
746 // Set slacks
747 z.r1 = 0.5*((z.mu - z.cE) + (z.cE.square() + z.mu*z.mu).sqrt());
748 z.r2 = 0.5*((z.mu + z.cE) + (z.cE.square() + z.mu*z.mu).sqrt());

752 z.r2.cwiseMax(p.slack_min);

755 // Check for inequality constraints
756 if (i.nI > 0)

758 // Set slacks

760 z.s2 = 0.5*((2.0*z.mu + z.cI) + (z.cI.square() + 4.0*z.mu*z.mu).sqrt());

762 // Adjust for numerical error
763 z.s1.cwiseMax(p.slack_min);
764 z.s2.cwiseMax(p.slack_min);

911 template <typename Real>
912 void Solver<Real>::updateParameters()
913 {
914
915 Parameter<Real> & p{this->m_parameter};
916 Iterate<Real> & z{this->m_iterate};
917
918 // Check for interior-point parameter update based on optimality error
919 while (z.mu > p.mu_min && z.kkt(2) <= std::max(z.mu, p.opt_err_tol-z.mu))
920 {
921 // Restrict interior-point parameter increase
922 this->setMuMaxExpZero();
923
924 // Update interior-point parameter
925 if (z.mu > p.mu_min)
926 {
927 // Decrease interior-point
928 z.mu = std::max(p.mu_min, std::min(p.mu_factor*z.mu, std::pow(z.mu, p.mu_factor_exp)));
929
930 // Evaluate penalty and interior-point parameter dependent quantities
931 this->evalDependent();
932 }
933 }
934
935 // Check for penalty parameter update based on optimality error
936 if ((z.kkt(1) <= p.opt_err_tol && z.v > p.opt_err_tol) ||
937 z.v > std::max({1.0, z.v_, p.infeas_max}))
938 {
939 // Update penalty parameter
940 if (z.rho > p.rho_min)
941 {
942 // Decrease penalty parameter
943 z.rho = std::max(p.rho_min, p.rho_factor*z.rho);
944
945 // Evaluate penalty and interior-point parameter dependent quantities
946 this->evalDependent();
947 }
948 }
