41 inline void backtracking(Acceptance<Real> & a, Parameter<Real> & p, Input<Real> & i, Counter & c,
42 Iterate<Real> & z, Direction<Real> const & d, Problem<Real> const & problem)
43 {
44 static constexpr Real EPSILON{std::numeric_limits<Real>::epsilon()};
45
46 // Store current values
47 Real f{z.f}, phi{z.phi};
48 Array<Real> cE(z.cE), r1(z.r1), r2(z.r2), lE(z.lE), cI(z.cI), s1(z.s1), s2(z.s2), lI(z.lI);
49 Vector<Real> x(z.x);
50
51 // Backtracking loop
52 while (a.p >= EPSILON)
53 {
54 // Set trial point
55 updatePoint<Real>(z, i, d, a);
56 evalFunctions<Real>(z, i, c, problem);
57
58 // Check for function evaluation error
59 if (z.err == 0)
60 {
61 // Set remaining trial values
62 evalSlacks<Real>(z, p, i);
63 evalMerit<Real>(z, i);
64
65 // Check for nonlinear fraction-to-boundary violation
66 Integer ftb{0};
67 const Real tmp_min{std::min(p.ls_frac, z.mu)};
68 if (i.nE > 0) {
69 ftb += (z.r1 < tmp_min*r1).count() + (z.r2 < tmp_min*r2).count();
70 }
71 if (i.nI > 0) {
72 ftb += (z.s1 < tmp_min*s1).count() + (z.s2 < tmp_min*s2).count();
73 }
74
75 // Check Armijo condition
76 if (ftb == 0 && z.phi - phi <= -p.ls_thresh*a.p*std::max(d.qtred, 0.0))
77 {
78 // Reset variables and return
79 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, z.f, z.cE, z.cI, z.phi);
80 return;
81 }
82 else
83 {
84 // Reset variables
85 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
86 }
87 }
88 else
89 {
90 // Reset variables
91 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
92 }
93
94 // Reduce step length
95 a.p *= p.ls_factor;
96 }

112 template <typename Real>
113 inline void fractionToBoundary(Acceptance<Real> & a, Parameter<Real> & p, Input<Real> const & i,
114 Iterate<Real> & z, Direction<Real> & d)
115 {
116 // Initialize primal fraction-to-boundary
117 a.p0 = 1.0;
118
119 // Update primal fraction-to-boundary for constraint slacks
120 Real frac{std::min(p.ls_frac, z.mu)};
121 if (i.nE > 0) {
122 const Indices idx_r1(find(d.r1 < 0.0)), idx_r2(find(d.r2 < 0.0));
123 Real min_1{std::numeric_limits<Real>::infinity()},
124 min_2{std::numeric_limits<Real>::infinity()};
125 if (idx_r1.size() > 0) {min_1 = (((frac - 1.0)*z.r1(idx_r1))/d.r1(idx_r1)).minCoeff();}
126 if (idx_r2.size() > 0) {min_2 = (((frac - 1.0)*z.r2(idx_r2))/d.r2(idx_r2)).minCoeff();}
127 a.p0 = std::min(a.p0, std::min(min_1, min_2));
128 }
129 if (i.nI > 0) {
130 const Indices idx_s1(find(d.s1 < 0.0)), idx_s2(find(d.s2 < 0.0));
131 Real min_1{std::numeric_limits<Real>::infinity()},
132 min_2{std::numeric_limits<Real>::infinity()};
133 if (idx_s1.size() > 0) {min_1 = (((frac - 1.0)*z.s1(idx_s1))/d.s1(idx_s1)).minCoeff();}
134 if (idx_s2.size() > 0) {min_2 = (((frac - 1.0)*z.s2(idx_s2))/d.s2(idx_s2)).minCoeff();}
135 a.p0 = std::min(a.p0, std::min(min_1, min_2));
136 }
137
138 // Initialize primal step length
139 a.p = a.p0;
140
141 // Initialize dual fraction-to-boundary
142 a.d = 1.0;
143
144 // Update dual fraction-to-boundary for constraint multipliers
145 if (i.nE > 0) {
146 const Indices idx_l(find(d.lE < 0.0)), idx_g(find(d.lE > 0));
147 Real min_1{std::numeric_limits<Real>::infinity()},
148 min_2{std::numeric_limits<Real>::infinity()};
149 if (idx_l.size() > 0) {min_1 = (((frac - 1.0)*(1.0 + z.lE(idx_l)))/d.lE(idx_l)).minCoeff();}
150 if (idx_g.size() > 0) {min_2 = (((1.0 - frac)*(1.0 - z.lE(idx_g)))/d.lE(idx_g)).minCoeff();}
151 a.d = std::min(a.d, std::min(min_1, min_2));
152 }
153 if (i.nI > 0) {
154 const Indices idx_l(find(d.lI < 0.0)), idx_g(find(d.lI > 0));
155 Real min_1{std::numeric_limits<Real>::infinity()},
156 min_2{std::numeric_limits<Real>::infinity()};
157 if (idx_l.size() > 0) {min_1 = (((frac - 1.0)* z.lI(idx_l)) /d.lI(idx_l)).minCoeff();}
158 if (idx_g.size() > 0) {min_2 = (((1.0 - frac)*(1.0 - z.lI(idx_g)))/d.lI(idx_g)).minCoeff();}
159 a.d = std::min(a.d, std::min(min_1, min_2));
160 }
161 }

181 inline Integer fullStepCheck(Acceptance<Real> const & a, Parameter<Real> & p, Input<Real> & i,
182 Counter & c, Iterate<Real> & z, Direction<Real> const & d, Problem<Real> const & problem)
183 {
184 // Set current and last penalty parameters
185 Real rho{z.rho}, rho_temp{z.rho_};
186
187 // Loop through last penalty parameters
188 while (rho < rho_temp)
189 {
190 // Set penalty parameter
191 setRho<Real>(z, rho_temp);
192
193 // Evaluate merit
194 evalMerit<Real>(z, i);
195
196 // Store current values
197 Real phi{z.phi}, f{z.f};
198 Array<Real> cE(z.cE), r1(z.r1), r2(z.r2), lE(z.lE), cI(z.cI), s1(z.s1), s2(z.s2), lI(z.lI);
199 Vector<Real> x(z.x);
200
201 // Set trial point
202 updatePoint<Real>(z, i, d, a);
203 evalFunctions<Real>(z, i, c, problem);
204
205 // Check for function evaluation error
206 if (z.err == 0)
207 {
208 // Set remaining trial values
209 evalSlacks<Real>(z, p, i);
210 evalMerit<Real>(z, i);
211
212 // Check for nonlinear fraction-to-boundary violation
213 Integer ftb{0};
214 const Real tmp_min{std::min(p.ls_frac, z.mu)};
215 if (i.nE > 0) {
216 ftb += (z.r1 < tmp_min*r1).count() + (z.r2 < tmp_min*r2).count();
217 }
218 if (i.nI > 0) {
219 ftb += (z.s1 < tmp_min*s1).count() + (z.s2 < tmp_min*s2).count();
220 }
221
222 // Check Armijo condition
223 if (ftb == 0 && z.phi - phi <= -p.ls_thresh*a.p*std::max(d.qtred, 0.0))
224 {
225 // Reset variables, set boolean, and return
226 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, z.f, z.cE, z.cI, z.phi);
227 return 1;
228 }
229 else
230 {
231 // Reset variables
232 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
233 }
234 }
235 else
236 {
237 // Reset variables
238 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
239 }
240
241 // Decrease rho
242 rho_temp *= p.rho_factor;
243 }
244
245 // Set rho
246 setRho<Real>(z, rho);
247
248 // Evaluate merit
249 evalMerit<Real>(z, i);
250
251 return 0;
252 }

269 template <typename Real>
270 inline void lineSearch(Acceptance<Real> & a, Parameter<Real> & p, Input<Real> & i, Counter & c,
271 Iterate<Real> & z, Direction<Real> & d, Problem<Real> const & problem)
272 {
273 // Check fraction-to-boundary rule
274 fractionToBoundary<Real>(a, p, i, z, d);
275
276 // Check for full step for trial penalty parameters
277 Integer b{fullStepCheck<Real>(a, p, i, c, z, d, problem)};
278 // Run second-order correction
279 a.s = false;
280 if (b == 0) {
281 b = secondOrderCorrection<Real>(a, p, i, c, z, d, problem);
282 if (b == 2) {a.s = true;}
283 }
284
285 // Run backtracking line search
286 if (b == 0) {backtracking<Real>(a, p, i, c, z, d, problem);}
287 }

306 inline Integer secondOrderCorrection(Acceptance<Real> & a, Parameter<Real> & p, Input<Real> & i,
307 Counter & c, Iterate<Real> & z, Direction<Real> & d, Problem<Real> const & problem)
308 {
309 // Store current iterate values
310 Real f{z.f}, phi{z.phi}, v{z.v};
311 Array<Real> cE(z.cE), r1(z.r1), r2(z.r2), lE(z.lE), cI(z.cI), s1(z.s1), s2(z.s2), lI(z.lI);

320 {
321 // Set remaining trial values
322 evalSlacks<Real>(z, p, i);
323 evalMerit<Real>(z, i);
324
325 // Check for nonlinear fraction-to-boundary violation
326 Integer ftb{0};
327 const Real tmp_min{std::min(p.ls_frac, z.mu)};
328 if (i.nE > 0) {
329 ftb += (z.r1 < tmp_min*r1).count() + (z.r2 < tmp_min*r2).count();
330 }
331 if (i.nI > 0) {
332 ftb += (z.s1 < tmp_min*s1).count() + (z.s2 < tmp_min*s2).count();
333 }
334
335 // Check Armijo condition
336 if (ftb == 0 && z.phi - phi <= -p.ls_thresh*a.p*std::max(d.qtred, 0.0))
337 {
338 // Reset variables, set flag, and return
339 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, z.f, z.cE, z.cI, z.phi);
340 return 1;
341 }
342 else if (evalViolation<Real>(i, z.cE, z.cI) < v)
343 {
344 // Reset variables and return
345 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
346 return 0;
347 }
348 else
349 {
350 // Reset variables (but leave constraint values for second-order correction)
351 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, z.cE, z.cI, phi);
352 }
353 }
354 else
355 {
356 // Reset variables and return
357 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
358 return 0;
359 }
360
361 // Recompute slacks for second order correction
362 evalSlacks<Real>(z, p, i);
363
364 // Evaluate trial primal-dual right-hand side vector
365 evalNewtonRhs<Real>(z, i);
366
367 // Store current direction values
368 Vector<Real> dx(d.x), dr1(d.r1), dr2(d.r2), dlE(d.lE), ds1(d.s1), ds2(d.s2), dlI(d.lI);

370
371 // Evaluate search direction
372 evalNewtonStep<Real>(d, i, z);
373
374 // Set trial direction
375 setDirection<Real>(d, i, a.p*dx+d.x, a.p*dr1+d.r1.matrix(), a.p*dr2+d.r2.matrix(), a.p*ds1+d.s1.matrix(),
376 a.p*ds2+d.s2.matrix(), a.d*dlE+d.lE.matrix(), a.d*dlI+d.lI.matrix(), (a.p*dx+d.x).norm(),
377 std::sqrt((a.d*dlE+d.lE.matrix()).matrix().squaredNorm() + (a.d*dlI+d.lI.matrix()).squaredNorm()));
378
379 // Set trial point
380 updatePoint<Real>(z, i, d, a);
381 evalFunctions<Real>(z, i, c, problem);
382
383 // Check for function evaluation error
384 if (z.err == 0)
385 {
386 // Set remaining trial values
387 evalSlacks<Real>(z, p, i);
388 evalMerit<Real>(z, i);
389
390 // Check for nonlinear fraction-to-boundary violation
391 Integer ftb{0};
392 const Real tmp_min{std::min(p.ls_frac, z.mu)};
393 if (i.nE > 0) {
394 ftb += (z.r1 < tmp_min*r1).count() + (z.r2 < tmp_min*r2).count();
395 }
396 if (i.nI > 0) {
397 ftb += (z.s1 < tmp_min*s1).count() + (z.s2 < tmp_min*s2).count();
398 }
399
400 // Check Armijo condition
401 if (ftb == 0 && z.phi - phi <= -p.ls_thresh*a.p*std::max(d.qtred, 0.0))
402 {
403 // Reset variables, set flag, and return
404 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, z.f, z.cE, z.cI, z.phi);
405 return 2;
406 }
407 else
408 {
409 // Reset variables
410 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);
411 }
412 }

415 // Reset variables
416 setPrimals<Real>(z, i, x, r1, r2, s1, s2, lE, lI, f, cE, cI, phi);

420 setDirection(d, i, dx, dr1, dr2, ds1, ds2, dlE, dlI, dx_norm, dl_norm);
421
422 // Reduce step length
423 a.p *= p.ls_factor;
