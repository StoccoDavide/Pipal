151 void printFooter(Parameter<Real> const & p, Input<Real> const & i, Counter const & c,
152 Iterate<Real> const & z) const
153 {
154 this->printIterate(c, z);
155 this->s
156 << this->n << std::endl << this->l << std::endl
157 << std::endl;
158
159 const Integer b{checkTermination(z, p, i, c)};
160
161 this->s
162 << "Final result" << std::endl
163 << "============" << std::endl;
164 switch (b) {
165 case 0: this->s << " EXIT: No termination message set" << std::endl; break;
166 case 1: this->s << " EXIT: Optimal solution found" << std::endl; break;
167 case 2: this->s << " EXIT: Infeasible stationary point found" << std::endl; break;
168 case 3: this->s << " EXIT: Iteration limit reached" << std::endl; break;
169 case 4: this->s << " EXIT: Invalid bounds" << std::endl; break;
170 case 5: this->s << " EXIT: Function evaluation error" << std::endl; break;
171 default: this->s << " EXIT: Unknown termination" << std::endl; break;
172 }
173
174 this->s
175 << "\nFinal values" << std::endl
176 << "============" << std::showpos << std::endl
177 << " Objective function........................ : " << z.fu << std::endl
178 << " Feasibility violation..................... : " << z.vu << std::endl
179 << " Optimality error (feasibility)............ : " << z.kkt(0) << std::endl
180 << " Optimality error (penalty)................ : " << z.kkt(1) << std::endl
181 << " Optimality error (penalty-interior-point). : " << z.kkt(2) << std::endl
182 << " Penalty parameter......................... : " << z.rho << std::endl
183 << " Interior-point parameter.................. : " << z.mu << std::noshowpos << std::endl
184 << std::endl;
185
186 this->s
187 << "Final counters" << std::endl
188 << "==============" << std::endl
189 << " Iterations................................ : " << c.k << std::endl
190 << " Function evaluations...................... : " << c.f << std::endl
191 << " Gradient evaluations...................... : " << c.g << std::endl
192 << " Hessian evaluations....................... : " << c.H << std::endl
193 << " Matrix factorizations..................... : " << c.M << std::endl
194 << " CPU millseconds........................... : " << std::scientific << std::setprecision(4)
195 << std::chrono::duration_cast<MicroSeconds>(SteadyClock::now() - this->t).count()/1.0e3 << std::endl;
196 }
